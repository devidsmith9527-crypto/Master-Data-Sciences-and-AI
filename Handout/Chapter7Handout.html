១. សេចក្តីផ្តើមអំពី Searching (Linear vs Binary Search)
Searching (ការស្វែងរកទិន្នន័យ) គឺជាកិច្ចការមូលដ្ឋានដ៏សំខាន់បំផុតមួយនៅក្នុងវិទ្យាសាស្ត្រកុំព្យូទ័រ ដែលត្រូវស្វែងរកទីតាំង (Index) នៃតម្លៃទិន្នន័យណាមួយ (Target) នៅក្នុងបញ្ជី (Array/List)។ ក្បួនដោះស្រាយស្វែងរកដែលប្រើប្រាស់ច្រើនជាងគេមានពីរគឺ៖
១. Linear Search (ការស្វែងរកម្តងមួយៗពីដើមដល់ចប់)

• គោលការណ៍៖ ដំណើរការឆែកមើលទិន្នន័យម្តងមួយៗពីដំបូងគេបង្អស់ (Index 0) រហូតដល់ចុងក្រោយគេបង្អស់។ បើទិន្នន័យមាន $n$ ធាតុ ក្នុងករណីអាក្រក់បំផុត (រកមិនឃើញ ឬនៅចុងគេ) ត្រូវឆែក $n$ ដង (ល្បឿន O(n))។
• ករណីសិក្សា៖ ប្រៀបដូចជាការដែលអ្នកដើរស្វែងរកផ្ទះរបស់មិត្តភក្តិនៅលើផ្លូវមួយដែលគ្មានលេខរៀងសម្គាល់ច្បាស់លាស់។ អ្នកគ្មានជម្រើសអ្វីក្រៅពីដើរគោះទ្វារសួរម្តងមួយផ្ទះចាប់ពីក្បាលផ្លូវរហូតដល់ចុងផ្លូវឡើយ។

២. Binary Search (ការស្វែងរកបែបពុះកណ្តាល)

• គោលការណ៍៖ ដំណើរការស្វែងរកដោយលោតទៅប្រៀបធៀបចំចំណុចកណ្តាល (Mid) ជានិច្ច។ បើតម្លៃដែលចង់រកធំជាង Mid នោះប្រព័ន្ធនឹងបោះបង់ទិន្នន័យនៅខាងឆ្វេងទាំងស្រុង និងបន្តពុះស្វែងរកខាងស្តាំ។ ដំណើរការច្រោះចោល ៥០% នៃទិន្នន័យរាល់ជុំនៃការប្រៀបធៀបនេះ ធ្វើឱ្យល្បឿនកើនឡើងរហ័សដល់កម្រិត O($\log n$)។
• ករណីសិក្សា៖ ប្រៀបដូចជាការស្វែងរកពាក្យនៅក្នុងវចនានុក្រម (ដែលរៀបតាមលំដាប់អក្សរ A-Z រួចស្រាប់)។ អ្នកនឹងបើកពាក់កណ្តាលសៀវភៅជានិច្ច បើពាក្យដែលចង់រកធំជាងទំព័របើកនោះ អ្នកនឹងបោះចោលទំព័រខាងឆ្វេងចោលភ្លាមៗ O($\log n$)។

លក្ខខណ្ឌតម្រូវស្នូលរបស់ Binary Search៖ Binary Search អាចដំណើរការបាន លុះត្រាតែទិន្នន័យត្រូវបានតម្រៀប (Sorted) រួចរាល់ជាមុនសិន។ ប្រសិនបើទិន្នន័យរាយប៉ាយខ្វាត់ខ្វែង វាប្រៀបដូចជាសៀវភៅវចនានុក្រមដែលច្របូកច្របល់លំដាប់អក្សរ ដែលយើងមិនអាចពុះស្វែងរកបានឡើយ។

២. គោលការណ៍ និងយន្តការចង្អុលរបស់ Binary Search
ដើម្បីដំណើរការ Binary Search នៅក្នុងកូដកុំព្យូទ័រ យើងត្រូវប្រើប្រាស់ **Pointer (ខ្សែចង្អុល)** ចំនួនបី គឺ L (Left - ឆ្វេងបំផុត), R (Right - ស្តាំបំផុត), និង Mid (Middle - ចំណុចកណ្តាល)។
ដំណើរការវិភាគលំដាប់លំដោយគណិតវិទ្យា

ជំហានទី១៖ កំណត់ Left = 0, Right = n - 1 (ក្នុងឧទាហរណ៍ n = 10 ដូចនេះ Right = 9)។
ជំហានទី២៖ គណនា Mid = (Left + Right) // 2 = (0 + 9) // 2 = 4។ តម្លៃនៅ arr[4] = 16។
ជំហានទី៣៖ ដោយសារតែ Target (23) ធំជាង 16 ដូចនេះយើងដឹងច្បាស់ថា Target គ្មានផ្លូវស្ថិតនៅចន្លោះ Left ដល់ Mid ឡើយ។
ជំហានទី៤៖ ដូរ Left = Mid + 1 = 5 (បន្តស្វែងរកចន្លោះ Index 5 ដល់ 9)។ ដំណើរការបន្តរហូតដល់រកឃើញ ឬ Left ដើរហួស Right (រកមិនឃើញ)។

៣. ការអនុវត្តកូដស្វែងរកបែប Binary Search ក្នុង Python
ខាងក្រោមនេះជាកូដ Python ពេញលេញក្នុងការអនុវត្តក្បួនស្វែងរក Binary Search (Iterative Method) ដោយប្រើប្រាស់ Loop while ដើម្បីដេញចាប់យកតម្លៃដែលចង់បាន O($\log n$)។
1_binary_search.pyBinary Search Algorithm
def binary_search(arr, target):
    left = 0
    right = len(arr) - 1

    while left <= right:
        mid = (left + right) // 2 # រកចំណុចកណ្តាលពាក់កណ្តាល

        # កៅអីកណ្តាលចំលេខគោលដៅ
        if arr[mid] == target:
            return mid
        
        # បើលេខគោលដៅធំជាង ត្រូវរត់ទៅរកខាងស្តាំ
        elif arr[mid] < target:
            left = mid + 1
        
        # បើលេខគោលដៅតូចជាង ត្រូវរត់ទៅរកខាងឆ្វេង
        else:
            right = mid - 1

    return -1 # រកមិនឃើញ

# ទិន្នន័យបានតម្រៀបរួចជាស្រេច (Sorted Array)
sorted_data = [2, 5, 8, 12, 16, 23, 38, 56, 72, 91]

print("លទ្ធផលការស្វែងរក Binary Search:")
idx1 = binary_search(sorted_data, 23)
print(f"ស្វែងរកលេខ 23: រកឃើញនៅ Index {idx1}")

idx2 = binary_search(sorted_data, 99)
print(f"ស្វែងរកលេខ 99: {idx2 if idx2 != -1 else 'រកមិនឃើញឡើយ'}")

១. Bubble Sort (ការតម្រៀបបែបពពុះទឹក O($n^2$))

• គោលការណ៍៖ ដំណើរការប្រៀបធៀបទិន្នន័យជាគូៗជាប់គ្នា។ ប្រសិនបើអ្នកនៅខាងមុខធំជាងអ្នកនៅខាងក្រោយ វានឹងធ្វើការប្តូរទីតាំងគ្នា (Swap)។ វាបន្តធ្វើបែបនេះម្តងហើយម្តងទៀតរហូតដល់គ្មានការដោះដូរទៀតឡើយ។
• ករណីសិក្សា៖ ប្រដូចទៅនឹងពពុះទឹកស្រាលៗ (លេខតូចៗ) ដែលបណ្តែតខ្លួនឡើងមកលើផ្ទៃទឹកសន្សឹមៗម្តងមួយជំហានៗ រីឯដុំថ្មធ្ងន់ៗ (លេខធំៗ) នឹងលិចចុះទៅបាតបឹងជម្រៅចុងក្រោយគេបង្អស់។

២. Insertion Sort (ការតម្រៀបបែបស៊កបញ្ចូល O($n^2$))

• គោលការណ៍៖ ដំណើរការតម្រៀបដោយបែងចែក Array ជាពីរផ្នែក (ផ្នែកតម្រៀបរួច និងផ្នែកមិនទាន់តម្រៀប)។ វាដកយកទិន្នន័យម្តងមួយៗពីផ្នែកមិនទាន់តម្រៀប យកមកប្រៀបធៀបរុញរំកិល រួចស៊កបញ្ចូលទៅក្នុងទីតាំងត្រឹមត្រូវ។
• ករណីសិក្សា៖ ប្រដូចទៅនឹងការដែលអ្នកលេងបៀរ ចាប់ទាញសន្លឹកបៀរថ្មីម្តងមួយសន្លឹកចេញពីលើតុ យកមកស៊កបញ្ចូលទៅក្នុងដៃរបស់អ្នកឱ្យត្រូវតាមលំដាប់លំដៅលេខរៀបយ៉ាងស្អាត O(n) ពេលទិន្នន័យស្ទើរតែតម្រៀបរួចស្រាប់។

ដែនកំណត់របស់ក្បួនតម្រៀបគ្រឹះ O($n^2$)

ទោះបីជា Bubble Sort និង Insertion Sort មានភាពងាយស្រួលយល់ និងសរសេរកូដក៏ដោយ ក៏ពួកវាមានល្បឿនយឺតខ្លាំងកម្រិត O($n^2$) (Quadratic Time) ក្នុងករណីទិន្នន័យធំៗ។ ប្រសិនបើមានទិន្នន័យ ១ លានតួ កុំព្យូទ័រត្រូវធ្វើការប្រៀបធៀបដល់ទៅ ១ ទ្រីលានដង ($1,000,000^2$) ធ្វើឱ្យប្រព័ន្ធកកស្ទះដំណើរការ (គាំងម៉ាស៊ីន)។ ដូចនេះគេត្រូវការក្បួនតម្រៀបកម្រិតខ្ពស់ជំនួសវិញ។

# ១. ក្បួន Bubble Sort O(n²)
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                # ធ្វើការ Swap ប្តូរទីតាំងគ្នា
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

# ២. ក្បួន Insertion Sort O(n²)
def insertion_sort(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        # រុញទិន្នន័យដែលធំជាង key ទៅមុខមួយកន្លែង
        while j >= 0 and key < arr[j]:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key # ស៊កបញ្ចូល key ក្នុងទីតាំងត្រឹមត្រូវ
    return arr

# ដំណើរការសាកល្បង
data1 = [64, 34, 25, 12, 22, 11, 90]
print(f"ទិន្នន័យដើម: {data1}")
print(f"លទ្ធផល Bubble Sort: {bubble_sort(data1.copy())}")

data2 = [64, 34, 25, 12, 22, 11, 90]
print(f"លទ្ធផល Insertion Sort: {insertion_sort(data2)}")

៦. ក្បួនតម្រៀបកម្រិតខ្ពស់បែប Divide & Conquer (ចែកដើម្បីគ្រប់គ្រង)
ដើម្បីលុបបំបាត់ភាពយឺតយ៉ាវ O($n^2$) នៃក្បួនដោះស្រាយគ្រឹះ វិស្វករ DSA បានប្រើប្រាស់យុទ្ធសាស្ត្រ Divide & Conquer (ការពុះបំបែក និងគ្រប់គ្រង)។ វាធ្វើការបំបែកបញ្ហាធំដ៏ច្របូកច្របល់ ទៅជាបញ្ហាតូចៗងាយស្រួលដោះស្រាយ រួចទើបផ្គុំលទ្ធផលតម្រៀបត្រឡប់មកវិញ ធ្វើឱ្យល្បឿនស្ទុះលឿនដល់កម្រិត O($n \log n$)។ ក្បួនដោះស្រាយល្បីៗមានពីរគឺ៖
១. Merge Sort (ការពុះពាក់កណ្តាល និងផ្គុំចូល O($n \log n$))

• គោលការណ៍៖ ពុះបំបែក Array ច្របូកច្របល់ជាពីរកំណាត់ស្មើៗគ្នាជាបន្តបន្ទាប់ រហូតដល់សល់តែធាតុទោលដាច់ដោយឡែក (ដែលតំណាងឱ្យធាតុដែលតម្រៀបរួចស្រាប់) រួចទើបហៅអនុគមន៍គូសផ្គុំ (Merge) តម្រៀបចូលគ្នាវិញតាមលំដាប់ឡើងលើ។
• ករណីសិក្សា៖ ដូចជានាយកប្រតិបត្តិ (CEO) ទទួលបានឯកសារច្របូកច្របល់ ១០០ សន្លឹក។ គាត់ពុះចែកការងារឱ្យជំនួយការ ២ នាក់ម្នាក់ ៥០ សន្លឹក។ កូនចៅទាំងពីររៀបចំឯកសាររួចរាល់យ៉ាងស្អាត រួចយកមកប្រគល់ឱ្យ CEO ផ្គុំបញ្ជូលគ្នាវិញ O(n)។

២. Quick Sort (ការតម្រៀបរហ័សដោយរើសបង្គោល O($n \log n$))

• គោលការណ៍៖ ជ្រើសរើសធាតុមួយធ្វើជាចំណុចបង្គោល Pivot (មេក្រុម)។ រួចរៀបចំទិន្នន័យ៖ ធាតុណាដែលតូចជាង Pivot ត្រូវរត់ទៅនៅខាងឆ្វេង ឯធាតុណាដែលធំជាង Pivot ត្រូវរត់ទៅនៅខាងស្តាំ។ បន្ទាប់មក ឆ្វេង និងស្តាំ បន្តរើស Pivot តូចៗដោះស្រាយបែបដដែលៗ (Recursion)។
• ករណីសិក្សា៖ ប្រៀបដូចជាគ្រូបង្រៀនកាយរឹទ្ធិដែលចង់តម្រៀបសិស្សតាមកម្ពស់។ គាត់រើសសិស្សម្នាក់ជាបង្គោល (Pivot) រួចឱ្យអ្នកដទៃប្រៀបធៀបកម្ពស់៖ នរណាទាបជាងបង្គោលទៅឈរឆ្វេង ខ្ពស់ជាងទៅឈរស្តាំ រួចបន្តធ្វើបែបនេះរហូតដល់សិស្សទាំងអស់រៀបជួរបានយ៉ាងស្អាត O($n \log n$)។

៧. ការអនុវត្តកូដ Quick Sort (Pivot Strategy) ក្នុង Python
ខាងក្រោមនេះជាកូដ Python ក្នុងការបង្កើតក្បួនតម្រៀប Quick Sort ដោយប្រើប្រាស់វិធីសាស្ត្រសរសេរបែបខ្លី និងងាយយល់ (Pythonic Method) តាមរយៈ List Comprehension ដើម្បីបំបែកទិន្នន័យជាបីក្រុម O($n \log n$)៖
3_quick_sort.pyQuick Sort Implementation
def quick_sort(arr):
    # លក្ខខណ្ឌបញ្ចប់ (Base Case): បើ Array មានតែមួយតួ ឬទទេ មិនបាច់តម្រៀបឡើយ
    if len(arr) <= 1:
        return arr

    # រើសយកធាតុចំកណ្តាលធ្វើជា Pivot (មេក្រុម)
    pivot = arr[len(arr) // 2]

    # ១. បង្កើតក្រុមឆ្វេង: រាល់គ្រប់លេខដែលតូចជាង Pivot
    left = [x for x in arr if x < pivot]

    # ២. បង្កើតក្រុមបម្រុងកណ្តាល: រាល់គ្រប់លេខដែលស្មើ Pivot
    middle = [x for x in arr if x == pivot]

    # ៣. បង្កើតក្រុមស្តាំ: រាល់គ្រប់លេខដែលធំជាង Pivot
    right = [x for x in arr if x > pivot]

    # ហៅអនុគមន៍ Quick Sort ខ្លួនឯងបែប Recursion លើក្រុមឆ្វេង និងស្តាំ រួចផ្គុំត្រឡប់មកវិញ
    return quick_sort(left) + middle + quick_sort(right)

# ដំណើរការសាកល្បង
unsorted_array = [10, 7, 8, 9, 1, 5]
print(f"ទិន្នន័យមុនតម្រៀប: {unsorted_array}")

sorted_array = quick_sort(unsorted_array)
print(f"ទិន្នន័យក្រោយតម្រៀប: {sorted_array}")
>_ Terminal Output

១១. សទ្ទានុក្រម និងពាក្យគន្លឹះប្រចាំជំពូកទី ៧ (Key Terms & Glossary)

• Linear Search (ស្វែងរកម្តងមួយ)៖ ក្បួនស្វែងរកសាមញ្ញដោយដើរឆែកមើលម្តងមួយៗពីដើមដល់ចុងបញ្ជី O(n) សាកសមសម្រាប់បញ្ជីទំហំតូច ឬមិនទាន់បានតម្រៀប។

• Binary Search (ស្វែងរកពុះកណ្តាល)៖ ក្បួនស្វែងរកដ៏មានប្រសិទ្ធភាពកម្រិត O($\log n$) ដោយធ្វើការឆែកចំណុចកណ្តាល និងច្រោះចោលទិន្នន័យ ៥០% រាល់ជុំនៃការប្រៀបធៀប។

• Divide and Conquer៖ យុទ្ធសាស្ត្រពុះបំបែកបញ្ហាធំៗជាបញ្ហាតូចៗងាយស្រួលដោះស្រាយ រួចទើបផ្គុំបញ្ចូលលទ្ធផលតម្រៀបត្រឡប់មកវិញ (ប្រើក្នុង Merge & Quick Sort)។

• Pivot (ចំណុចបង្គោល)៖ ធាតុដែលត្រូវបានជ្រើសរើសធ្វើជាមេក្រុមនៅក្នុង Quick Sort ដើម្បីធ្វើការបែងចែកក្រុមធំជាង និងក្រុមតូចជាងបង្គោលនោះ។

• Swap (ការប្តូរទីតាំង)៖ សកម្មភាពនៃការដោះដូរតម្លៃទីតាំងគ្នារវាងធាតុពីរក្បែរគ្នាក្នុង Array ប្រើប្រាស់ជាចម្បងក្នុង Bubble Sort O($n^2$)។

• In-place Sorting (ការតម្រៀបក្នុងបញ្ជីដើម)៖ ក្បួនតម្រៀបដែលមិនទាមទារបង្កើត Array ថ្មីដើម្បីផ្ទុកទិន្នន័យ (សន្សំសំចៃ RAM ដូចជា Quick Sort)។

ការថ្លឹងថ្លែងប្រសិទ្ធភាពក្នុងការសរសេរកូដជាក់ស្តែង

ក្នុងនាមជាវិស្វករ Software និង Data Scientist កម្រិតខ្ពស់ យើងត្រូវយល់ច្បាស់ពីយន្តការគណនា Big-O។ ទោះបីជា Python ផ្តល់ Built-in list.sort() ដ៏ឆ្លាតវៃ (ដែលដំណើរការដោយក្បួន Timsort - ការបញ្ចូលគ្នារវាង Merge និង Insertion Sort) ក៏ដោយ ក៏ការយល់ច្បាស់ពីយន្តការខាងក្រោយជួយឱ្យយើងអាចរចនាក្បួនដោះស្រាយពិសេសផ្ទាល់ខ្លួន សម្រេចល្បឿនដំណើរការប្រព័ន្ធបានយ៉ាងរហ័សបំផុត និងចៀសវាងការខ្ជះខ្ជាយធនធាន RAM របស់កុំព្យូទ័រជាអតិបរមា។
