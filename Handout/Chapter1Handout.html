  • យល់ដឹងស៊ីជម្រៅ៖ ពីសារៈសំខាន់នៃ Data Structures និង Algorithms នៅក្នុងការអភិវឌ្ឍប្រព័ន្ធបញ្ញាសិប្បនិម្មិត (AI) និងវិទ្យាសាស្ត្រទិន្នន័យខ្នាតធំ (Big Data)។

១. ហេតុអ្វីវិស័យបញ្ញាសិប្បនិម្មិត (AI) ត្រូវការ DSA?
ក្នុងយុគសម័យ Big Data កម្មវិធីបញ្ញាសិប្បនិម្មិត (Artificial Intelligence - AI) ត្រូវស្រូបយកទិន្នន័យរាប់លាន (Terabytes/Petabytes) ដើម្បីធ្វើការបង្វឹក (Training)។ បើយើងមិនមានការរៀបចំទិន្នន័យ (Data Structure) ឱ្យបានត្រឹមត្រូវ និងខ្វះក្បួនដោះស្រាយ (Algorithm) ដែលមានល្បឿនលឿនទេនោះ ការបង្វឹកម៉ូដែលយន្តកម្មសិក្សា (Machine Learning - ML) អាចចំណាយពេលរាប់ខែ ឬធ្វើឲ្យគាំងម៉ាស៊ីន (Out of Memory)។ ការសិក្សា DSA កម្រិតខ្ពស់ គឺជាការរៀនពីរបៀបសន្សំសំចៃពេលវេលា និងធនធានម៉ាស៊ីនយ៉ាងមានប្រសិទ្ធភាពបំផុត។
២. និយមន័យស៊ីជម្រៅ៖ Data Structures

តើអ្វីទៅជា Data Structures?

គឺជាទម្រង់នៃស្ថាបត្យកម្មកូដ ដែលត្រូវបានប្រើសម្រាប់រៀបចំ និងរក្សាទុកទិន្នន័យក្នុងអង្គចងចាំចម្បង (Random Access Memory - RAM) ដើម្បីឱ្យកុំព្យូទ័រអាចទាញយក (Access) បញ្ចូលថ្មី (Insert) ឬលុបចោល (Delete) បានលឿនបំផុតទៅតាមតម្រូវការជាក់ស្តែង។

ឧទាហរណ៍៖ ទូទឹកកក vs ឃ្លាំងទិន្នន័យ

បើយើងទិញវត្ថុធាតុដើមមកគរទុកក្នុងថង់តែមួយ ការស្វែងរកសាច់ជ្រូកទាមទារការរើកកាយទាំងអស់ O(n)។ ការមាន "ថតទូទឹកកក" ដាច់ដោយឡែក (Arrays, Hash Maps, Trees) អនុញ្ញាតឲ្យចាប់យករបស់ដែលចង់បានភ្លាមៗ O(1)។ ទិន្នន័យរាប់លានជួរក្នុងមូលដ្ឋានទិន្នន័យ (Database - DB) ក៏ពឹងផ្អែកលើនេះដែរ។

៣. និយមន័យស៊ីជម្រៅ៖ Algorithms

តើអ្វីទៅជា Algorithms?

គឺជាបណ្តុំនៃជំហាន តក្កវិជ្ជា និងរូបមន្តគណិតវិទ្យា ដែលប្រាប់កុំព្យូទ័រពី "របៀប" ក្នុងការដោះស្រាយបញ្ហា។ ក្នុងបញ្ញាសិប្បនិម្មិត (AI) ក្បួនដូចជា Backpropagation គឺជាក្បួនសម្រាប់កែតម្រូវកំហុសរបស់ Neural Networks រហូតទាល់តែបានចម្លើយត្រឹមត្រូវ។

ឧទាហរណ៍៖ ការរកផ្លូវលើផែនទី

ផែនទីមានទិន្នន័យផ្លូវខ្វាត់ខ្វែងជារាង Graph (Data Structure)។ ពេលចង់ទៅគោលដៅ Google មិនអាចបើកឡានសាកល្បងគ្រប់ផ្លូវទេ។ គេប្រើប្រាស់ Dijkstra's Algorithm ដើម្បីគណនារកផ្លូវដែលចំណាយពេលតិចបំផុតត្រឹមតែប៉ុន្មានមិល្លីវិនាទី។

សរុប៖ Data Structure គឺជាឃ្លាំងផ្ទុកអាវុធ ចំណែក Algorithm គឺជាយុទ្ធសាស្ត្រក្នុងការប្រើប្រាស់អាវុធទាំងនោះដើម្បីយកឈ្នះសង្គ្រាម (ដោះស្រាយបញ្ហា)។

ក្នុងសង្គមជាក់ស្តែង៖ សៀវភៅធ្វើម្ហូប

៥. ការវិភាគប្រសិទ្ធភាពក្បួនដោះស្រាយ (Complexity Analysis)
តើយើងដឹងថាកូដមួយណា "ល្អជាង" កូដមួយណាដោយរបៀបណា? យើងមិនអាចវាស់ជា "វិនាទី" (Seconds) បានទេ ព្រោះកុំព្យូទ័រធំៗ (Server) ដើរលឿនជាងកុំព្យូទ័រយួរដៃ (Laptop)។ ដូច្នេះ វិស្វករកុំព្យូទ័រប្រើប្រាស់រង្វាស់ Complexity Analysis ដើម្បីវាស់ពីរបៀបដែលកូដទាមទារធនធាន ពេលដែលទិន្នន័យ (n) កើនឡើងធំទៅៗ។

Time Complexity (ល្បឿន)

វាស់ស្ទង់ថាតើ ក្បួនដោះស្រាយនេះត្រូវប្រើពេលយូរប៉ុណ្ណា (ឬប៉ុន្មានជំហាន) ពេលទិន្នន័យ (n) កើនឡើង។

ក្នុងវិស័យ NLP

ម៉ូដែលយល់ដឹងភាសា (NLP) ដែលមាន Time Complexity ខ្ពស់ អាចទាមទារពេល ៣ខែក្នុងការ Train។ ការកាត់បន្ថយ Time Complexity អាចឲ្យវា Train ចប់ត្រឹម ៣ថ្ងៃ។

Space Complexity (អង្គចងចាំ)

វាស់ស្ទង់ថាតើ ក្បួនដោះស្រាយនេះទាមទារទំហំអង្គចងចាំ (RAM) ប៉ុន្មានសម្រាប់ស្តុកទុកទិន្នន័យបណ្តោះអាសន្ន។

ក្នុងវិស័យ Computer Vision

បើយើងផ្ទុករូបភាព ១លានសន្លឹកចូល RAM ព្រមគ្នា Server នឹងគាំងភ្លាម (Out of memory)។ យើងត្រូវបំបែកជាតំណាក់កាលដើម្បីសន្សំ Space។

៦. ទិដ្ឋភាពទូទៅនៃ Big-O Notation និង Worst-Case
Big-O Notation គឺជាភាសាគណិតវិទ្យាប្រើសម្រាប់តំណាងឲ្យ Complexity ។ អក្សរ "O" មកពីពាក្យ "Order of Magnitude"។ យើងតែងតែគណនា Big-O ដោយផ្តោតលើ Worst-case Scenario (ករណីដែលកូដត្រូវប្រើពេលយូរបំផុត)។
ហេតុអ្វីយើងវាស់ Worst-case?

ឧបមាថាអ្នករកឈ្មោះ "Zebra" ក្នុងបញ្ជីសត្វ ១លាន។ បើវានៅលេខរៀងទី១ វាលឿនបំផុត (Best-case)។ តែយើងត្រូវប្រាកដថាប្រព័ន្ធមិនគាំង ទោះបីជាវានៅលេខរៀងទី ១លាន ឬក៏គ្មានសោះក៏ដោយ (Worst-case)។ យើងតែងត្រៀមលក្ខណៈសម្រាប់ករណីអាក្រក់បំផុតជានិច្ច!

ចំណាត់ថ្នាក់ល្បឿន (ពីលឿនទៅយឺត)

  • O(1): លឿនថេរជានិច្ច (Excellent)
  • O(log n): លឿនខ្លាំង (Good)
  • O(n): ល្បឿនមធ្យម ស្របតាមទិន្នន័យ (Fair)
  • O(n log n): ស្តង់ដារនៃការរៀបចំ (Sorting)
  • O(n²): យឺតខ្លាំង (Terrible - គួរជៀសវាង)

2_logarithmic_time.py (Binary Search)O(log n)
def binary_search(sorted_arr, target):
    left, right = 0, len(sorted_arr) - 1
    steps = 0 # រាប់ចំនួនជំហានដែលម៉ាស៊ីនបានដើរ
    
    while left <= right:
        steps += 1
        mid = (left + right) // 2 # រកចំណុចកណ្តាល
        
        if sorted_arr[mid] == target:
            return mid, steps
        elif sorted_arr[mid] < target:
            left = mid + 1  # កាត់ពាក់កណ្តាលខាងឆ្វេងចោល
        else:
            right = mid - 1 # កាត់ពាក់កណ្តាលខាងស្តាំចោល
            
    return -1, steps

# បង្កើតបញ្ជីលេខរៀងពី 1 ដល់ 1,000,000
huge_data = list(range(1, 1000001))
target_number = 850420

index, step_count = binary_search(huge_data, target_number)
print(f"ស្វែងរកលេខ: {target_number} ក្នុងចំណោមទិន្នន័យ 1 លាន")
print(f"រកឃើញនៅ Index ទី: {index}")

៩. ការវិភាគល្បឿន O(n) - Linear Time

3_linear_time.pyO(n)
def find_max_value(dataset):
    max_val = dataset[0]
    steps = 0
    
    for item in dataset:  # Loop ដើរ n ដង តាមចំនួនទិន្នន័យជាក់ស្តែង
        steps += 1
        if item > max_val:
            max_val = item
            
    return max_val, steps

data = [45, 12, 99, 3, 78, 82, 1] # ទិន្នន័យ ៧ តួ (n=7)
max_result, step_count = find_max_value(data)

print(f"តម្លៃធំបំផុតគឺ: {max_result}")
print(f"ម៉ាស៊ីនបានដើរពិនិត្យចំនួន: {step_count} ជំហាន (O(n))")

១០. ការវិភាគល្បឿន O(n²) - Quadratic Time (គួរជៀសវាង)
និយមន័យ៖ ល្បឿនយឺតខ្លាំងបំផុត ដែលយើងគួរតែជៀសវាងឲ្យបានជាដាច់ខាត។ កើតឡើងនៅពេលដែលមានការប្រើប្រាស់ Loop ជាន់គ្នាពីរជាន់ (Nested Loops) ដើម្បីប្រៀបធៀបធាតុនីមួយៗជាមួយធាតុផ្សេងទៀតទាំងអស់ (ឧ. ការរកទិន្នន័យស្ទួនដោយមិនមានការរៀបចំជាមុន)។
4_quadratic_time_bad.pyO(n²)
import time

def find_duplicates_bad(dataset):
    n = len(dataset)
    steps = 0
    duplicates = []
    
    for i in range(n):          # Loop ទី១ ដើរ n ដង
        for j in range(n):      # Loop ទី២ ក៏ដើរ n ដង (ជាន់គ្នា)
            steps += 1
            # បើ Index ខុសគ្នា តែទិន្នន័យដូចគ្នា នោះវាជា Duplicate
            if i != j and dataset[i] == dataset[j] and dataset[i] not in duplicates:
                duplicates.append(dataset[i])
                
    return duplicates, steps

# សាកល្បងជាមួយទិន្នន័យ ១,០០០ ជួរ (n = 1000)
# 1000 x 1000 ម៉ាស៊ីននឹងត្រូវដើរ ១ លានជំហាន!
data = list(range(1000))
data.append(450) # បញ្ចូលទិន្នន័យស្ទួនចេតនា

start = time.time()
dupes, step_count = find_duplicates_bad(data)
end = time.time()

print(f"រកឃើញទិន្នន័យស្ទួន: {dupes}")
print(f"ម៉ាស៊ីនបានដើរចំនួន: {step_count:,} ជំហាន (O(n²))")
print(f"ចំណាយពេល: {end - start:.4f} វិនាទី")

១១. ច្បាប់នៃការលុបចោល (Dropping Constants in Big-O)
ពេលយើងសរសេរ Big-O យើងសរសេរតែភាពធំដុំ (Order of Magnitude) ប៉ុណ្ណោះ៖

បោះបង់មេគុណ (Drop Constants)

បើកូដមាន Loop ដើរ 2 ដង គឺ O(2n) យើងត្រូវលុបលេខ 2 ចោល ហើយសរសេរត្រឹម O(n)។ ព្រោះទោះដើរ 1 ឬ 2 ដង វានៅតែជាល្បឿនកម្រិត Linear ដដែល។

បោះបង់តួអន់ (Drop Non-dominant)

បើកូដមានល្បឿន O(n² + n) យើងលុប n ចោល សរសេរត្រឹម O(n²)។ ព្រោះនៅពេលទិន្នន័យរាប់លាន តម្លៃ n គឺតូចតាចពេកមិនមានឥទ្ធិពលអ្វីធំដុំឡើយ។

ការបកស្រាយ៖ Data Structures គឺជា "វិធីសាស្ត្រក្នុងការរៀបចំ និងស្តុកទុកទិន្នន័យ" (ប្រៀបដូចជាទូទឹកកកដែលមានថតត្រឹមត្រូវ) ចំណែកឯ Algorithms គឺជា "ជំហាន ឬក្បួនតក្កវិជ្ជាក្នុងការដោះស្រាយបញ្ហា" (ប្រៀបដូចជាសៀវភៅធ្វើម្ហូប)។ យើងត្រូវរៀនវាគួបផ្សំគ្នា ព្រោះ Algorithm ដែលល្អបំផុត ក៏មិនអាចដំណើរការលឿនបានដែរ ប្រសិនបើទិន្នន័យមិនត្រូវបានរៀបចំក្នុង Data Structure ដ៏ត្រឹមត្រូវមួយនោះ។
២. ហេតុអ្វីបានជាក្នុងការវិភាគ Big-O Notation វិស្វករតែងតែផ្តោតទៅលើ "ករណីអាក្រក់បំផុត (Worst-case Scenario)"?
ការបកស្រាយ៖ ការផ្តោតលើ Worst-case គឺដើម្បីធានាថាប្រព័ន្ធ ឬកម្មវិធីកុំព្យូទ័រមិនគាំង ក្នុងស្ថានភាពដែលអាក្រក់បំផុត។ ឧទាហរណ៍៖ បើយើងកំពុងស្វែងរកឈ្មោះមនុស្សម្នាក់ក្នុងបញ្ជីឈ្មោះ ១លាននាក់ ហើយឈ្មោះនោះស្ថិតនៅលេខរៀងចុងក្រោយគេបង្អស់ ឬគ្មានសោះក្នុងបញ្ជី កូដរបស់យើងត្រូវតែអាចទប់ទល់ និងដំណើរការវាបានដោយជោគជ័យ។ ការដឹងពីដែនកំណត់អាក្រក់បំផុតនេះ ជួយឱ្យយើងរចនាប្រព័ន្ធបានរឹងមាំ (Robust System)។

៣. សូមប្រៀបធៀបល្បឿនកម្រិត O(1) និង O(n) ដោយលើកយកឧទាហរណ៍ជាក់ស្តែងមកបញ្ជាក់។
ការបកស្រាយ៖
• O(1) - Constant Time: គឺជាល្បឿនថេរ ដែលមិនប្រែប្រួលទោះទិន្នន័យមានចំនួនច្រើនប៉ុណ្ណាក៏ដោយ។ ឧទាហរណ៍៖ ការទាញយកឈ្មោះសិស្សពី Dictionary ក្នុង Python ដោយស្គាល់ ID របស់គាត់។ កុំព្យូទ័រប្រើពេលតែ១ជំហានក្នុងការលោតទៅយកទិន្នន័យនោះ។
• O(n) - Linear Time: គឺជាល្បឿនដែលកើនឡើងស្របតាមចំនួនទិន្នន័យ។ ឧទាហរណ៍៖ ការស្វែងរកតម្លៃធំបំផុតក្នុងបញ្ជីពិន្ទុសិស្ស ១ពាន់នាក់ កុំព្យូទ័រត្រូវប្រើ For Loop ដើម្បីដើរឆែកមើលពិន្ទុរបស់សិស្សម្តងម្នាក់ៗទាំង ១ពាន់នាក់នោះទើបអាចដឹងលទ្ធផល។
៤. ហេតុអ្វីបានជាល្បឿន O(n²) ត្រូវបានចាត់ទុកថាមានគ្រោះថ្នាក់បំផុតសម្រាប់វិស័យបញ្ញាសិប្បនិម្មិត (AI)?
ការបកស្រាយ៖ វិស័យបញ្ញាសិប្បនិម្មិត (AI) ទាមទារការបង្វឹកលើទិន្នន័យខ្នាតធំ (Big Data)។ ល្បឿន O(n²) កើតឡើងនៅពេលមាន Loop ២ជាន់ (Nested Loops)។ ប្រសិនបើទិន្នន័យ (n) មានចំនួនត្រឹមតែ ១ លាន, នោះ O(n²) នឹងបង្ខំឱ្យម៉ាស៊ីន (Server) ធ្វើការគណនារហូតដល់ទៅ ១ ទ្រីលានដង (1,000,000 x 1,000,000)។ ដំណើរការនេះនឹងធ្វើឱ្យម៉ាស៊ីនគាំង ឬត្រូវចំណាយពេលរាប់សប្តាហ៍ទើបបញ្ចប់ ដែលធ្វើឱ្យខាតបង់ថវិកាយ៉ាងច្រើនលើសេវាកម្ម Cloud។
១៣. សទ្ទានុក្រម និងពាក្យគន្លឹះ (Glossary)

Asymptotic Analysis: ការវិភាគគណិតវិទ្យាដើម្បីវាយតម្លៃល្បឿនកូដ ពេលទិន្នន័យកើនឡើងដល់អនន្ត។

Trade-off: ការសម្រេចចិត្តលះបង់ចំណុចល្អមួយ ដើម្បីយកចំណុចល្អមួយទៀត (ឧ. លះបង់ RAM ដើម្បីល្បឿន)។

In-place Algorithm: ក្បួនដោះស្រាយដែលផ្លាស់ប្តូរទិន្នន័យលើ Memory ចាស់ ដោយមិនបង្កើត Array ថ្មី។

Optimization: ដំណើរការកែប្រែកូដ ឬទម្រង់ Machine Learning ឱ្យរត់លឿន និងស៊ីធនធានតិច។

១៤. សេចក្តីសន្និដ្ឋាន (Conclusion)
  • គ្រឹះនៃ AI: ម៉ូដែល AI ក៏ត្រូវបានកសាងលើគ្រឹះ DSA ដែរ។ បើគ្មាន DSA ការវិភាគ Big Data មិនអាចទៅរួចទេ។
  • ប្រើប្រាស់ Big-O: មុនសរសេរកូដ សូមសួរថា តើកូដនេះស្ថិតក្នុង O(n) ឬ O(n²)? វាជួយសង្គ្រោះថ្លៃ Cloud Computing។
