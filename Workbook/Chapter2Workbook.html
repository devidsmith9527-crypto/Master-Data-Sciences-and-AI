class BankAccount:
    def __init__(self):
        self.__balance = 0
        
    def deposit(self, amount):
        if amount > 0:
            self.__balance += amount

acc = BankAccount()

class Temperature:
    def __init__(self, c):
        self.__celsius = c

    @property
    def celsius(self): 
        return self.__celsius

    @celsius.setter
    def celsius(self, value):
        if value < -273.15: 
            raise ValueError("ត្រជាក់ពេក!")
        self.__celsius = value

t = Temperature(25)
t.celsius = 30 # ហៅ Setter ដូចអថេរធម្មតា
print("សីតុណ្ហភាព:", t.celsius)

មធ្យម ១ ការប្រើប្រាស់អនុគមន៍ super()

Copy & Run Code
class Person:
    def __init__(self, name):
        self.name = name

class Employee(Person):
    def __init__(self, name, salary):
        super().__init__(name) # ហៅ Constructor មេ
        self.salary = salary

emp = Employee("សុខ", 500)
print(f"បុគ្គលិក {emp.name} ប្រាក់ខែ {emp.salary}$")
អនុញ្ញាតឱ្យ Class កូនហៅ Constructor របស់ Class មេមកដំណើរការ ដើម្បីផ្ទុកតម្លៃ name ដោយកាត់បន្ថយការសរសេរកូដជាន់គ្នា។

ធ្ងន់ ១ ការបង្កើត Abstract Base Class (ABC)

Copy & Run Code
from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self):
        pass

class Square(Shape):
    def __init__(self, side):
        self.side = side
    def area(self):
        return self.side * self.side

s = Square(4)
print("ក្រឡាផ្ទៃ:", s.area())
Shape ជាពុម្ពគំរូអរូបី ដែលមិនអាចបង្កើត Object បាន។ វាបង្ខំឱ្យរាល់ Class កូនទាំងអស់ (ដូចជា Square) ត្រូវតែបង្កើត Method area() ជាដាច់ខាត។

def add_item(arr):
    arr.append(4)
    print("ក្នុង Function:", arr)

my_list = [1, 2, 3]
add_item(my_list)

មធ្យម ១ បញ្ហានៃការថតចម្លងសំបកក្រៅ (Shallow Copy)

import sys

៥. ករណីសិក្សា៖ Stack, Heap និង Garbage Collection (ត)

មធ្យម ១ ការកាត់ផ្តាច់ Reference ដោយប្រើ del

Copy & Run Code
a = ["Hello"]
b = a
del a # កាត់ខ្សែទី១ ចោល

try:
    print(b) # តើមាន Error ទេ?
except Exception as e:
    print(e)
បញ្ជា del a គ្រាន់តែលុបខ្សែតភ្ជាប់ (ស្លាកឈ្មោះ a) ចេញពី Stack ប៉ុណ្ណោះ។ ប្រអប់ទិន្នន័យពិត ["Hello"] នៅតែមានក្នុង Heap ព្រោះខ្សែ b នៅតែចង្អុលទៅវា។

ធ្ងន់ ១ ការបង្ខំឱ្យ Garbage Collection ដើរបោសសម្អាត

Copy & Run Code
import gc

class Node:
    pass

n1 = Node()
n2 = Node()

n1.ref = n2
n2.ref = n1 # ចង្អុលកន្ទុយគ្នាឯង (Cyclic)

del n1
del n2 # លុបឈ្មោះក្រៅ តែពួកវានៅចង្អុលគ្នា

collected = gc.collect() # បង្ខំ GC ឱ្យបោសសម្អាត
print("ចំនួន Object ដែលលុបចោល:", collected)
ក្នុងករណី Cyclic Reference ទោះបីយើង del ឈ្មោះចោលក៏ Reference Count មិនស្មើ 0 ដែរ។ ដូច្នេះត្រូវប្រើ gc.collect() ដើម្បីបង្ខំឱ្យ Python រាវរក និងកម្ទេចពួកវាចោលពី Memory។
