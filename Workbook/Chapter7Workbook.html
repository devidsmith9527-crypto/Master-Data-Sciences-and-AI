def linear_search(arr, target):
    for i in range(len(arr)):
        if arr[i] == target:
            return i  # រកឃើញ បោះទីតាំង Index ត្រឡប់
    return -1 # រកមិនឃើញ

លំហាត់ ៣ Iterative Binary Search (ការរាវរកពាក្យក្នុងវចនានុក្រមឌីជីថល)

Iterative Binary Search
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target: 
            return mid
        elif arr[mid] < target: 
            left = mid + 1
        else: 
            right = mid - 1
    return -1

words = ["Apple", "Banana", "Cherry", "Durian", "Grape", "Mango"]
result = binary_search(words, "Durian")
print("ទីតាំង Durian គឺ Index ទី:", result) # Expected Output: 3

def search_insert_position(grades, new_grade):
    left, right = 0, len(grades) - 1
    while left <= right:
        mid = (left + right) // 2
        if grades[mid] == new_grade: 
            return mid
        elif grades[mid] < new_grade: 
            left = mid + 1
        else: 
            right = mid - 1
    return left  # left ផ្ទុក index សមស្របសម្រាប់ស៊កបញ្ចូល

លំហាត់ ៦ Recursive Binary Search (ក្បួនស្វែងរកពុះកណ្តាលបែប Recursive)

Recursive Binary Search
def binary_search_rec(arr, target, left, right):
    if left > right: 
        return -1
    
    mid = (left + right) // 2
    if arr[mid] == target: 
        return mid
    elif arr[mid] < target:
        return binary_search_rec(arr, target, mid + 1, right)
    else:
        return binary_search_rec(arr, target, left, mid - 1)

numbers = [10, 20, 30, 40, 50, 60]
result = binary_search_rec(numbers, 50, 0, len(numbers) - 1)
print("ស្វែងរកលេខ 50 ឃើញនៅ Index ទី:", result) # Expected Output: 4

លំហាត់ ៧ Selection Sort (តម្រៀបទំហំឯកសាររូបថតពីតូចទៅធំ)

Selection Sort Code
def selection_sort(arr):
    n = len(arr)
    for i in range(n):
        min_idx = i
        for j in range(i + 1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j
        # Swap elements
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
    return arr

file_sizes = [45, 12, 89, 23, 11]
result = selection_sort(file_sizes)
print("ទំហំឯកសារតម្រៀបរួច:", result) # Expected Output: [11, 12, 23, 45, 89]

លំហាត់ ៩ Optimized Bubble Sort (តម្រៀបកម្ពស់សិស្សគោរពទង់ជាតិ)

Optimized Bubble Sort
def bubble_sort_optimized(arr):
    n = len(arr)
    for i in range(n):
        swapped = False
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        if not swapped: 
            break  # បញ្ឈប់ការរត់ loop ប្រសិនបើគ្មានការ Swap សោះ
    return arr

student_heights = [155, 170, 162, 150, 168]
result = bubble_sort_optimized(student_heights)
print("កម្ពស់តម្រៀបរួច:", result) # Expected Output: [150, 155, 162, 168, 170]

លំហាត់ ១១ Insertion Sort (ការតម្រៀបសន្លឹកបៀរលេងនៅលើដៃ)

Insertion Sort Code
def insertion_sort(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        # រុញធាតុដែលធំជាង key ទៅខាងស្តាំដើម្បីបើកផ្លូវ
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr

cards = [10, 2, 5, 3, 9]
result = insertion_sort(cards)
print("សន្លឹកបៀរតម្រៀបរួច:", result) # Expected Output: [2, 3, 5, 9, 10]

លំហាត់ ១៣ Merge Sort: Tracing ដំណើរការបំបែកទិន្នន័យ (Divide Phase)

កូដបង្ហាញគន្លងនៃការបំបែក (Divide) ទិន្នន័យជាចំណែកតូចៗរហូតដល់សល់ ១ ធាតុ៖

Divide Phase Tracing Code
def divide_trace(arr):
    if len(arr) <= 1: 
        return arr
    
    mid = len(arr) // 2
    left_half = arr[:mid]
    right_half = arr[mid:]
    
    print(f"ពុះចែក៖ {left_half} | {right_half}")
    divide_trace(left_half)
    divide_trace(right_half)

# សាកល្បងដំណើរការ tracing
divide_trace([101, 105, 103, 102])
# Expected Output printed:
# ពុះចែក៖ [101, 105] | [103, 102]
# ពុះចែក៖ [101] | [105]

លំហាត់ ១៤ Merge Arrays (ដំណើរការច្របាច់បញ្ចូល list ដែល Sorted ពីរចូលគ្នា)

កូដគំរូនៃ Merge Helper Function ដែលច្របាច់បញ្ចូលគ្នា៖

Merge Arrays Helper Function
def merge_arrays(left, right):
    res = []
    i = j = 0
    while i < len(left) and j < len(right):
        if left[i] < right[j]:
            res.append(left[i])
            i += 1
        else:
            res.append(right[j])
            j += 1
            
    # កើបយកទិន្នន័យដែលនៅសល់ (បើមាន)
    res.extend(left[i:])
    res.extend(right[j:])
    return res

# សាកល្បងដំណើរការ merging 
result = merge_arrays([101, 105], [102, 103])
print("ច្របាច់បញ្ចូលរួច:", result) # Expected Output: [101, 102, 103, 105]

កូដភ្ជាប់ទំនាក់ទំនងរវាងដំណើរការ Divide និង Helper function ដើម្បីតម្រៀបតម្លៃផលិតផល៖

Full Merge Sort Algorithm
def merge_sort(arr):
    if len(arr) <= 1: 
        return arr
    
    mid = len(arr) // 2
    # ដំណាក់កាលបំបែកទិន្នន័យ (Divide Phase)
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    
    # ដំណាក់កាលច្របាច់បញ្ចូល និងតម្រៀប (Conquer & Merge Phase)
    return merge_arrays(left, right) # ហៅ Helper Function ពីទំព័រទី ៥

prices = [45.5, 12.0, 89.9, 3.5, 25.0]
result = merge_sort(prices)
print("តម្លៃផលិតផលតម្រៀបរួច:", result) 
# Expected Output: [3.5, 12.0, 25.0, 45.5, 89.9]

៧. ក្បួនតម្រៀបរហ័ស Quick Sort (លំហាត់ ១៧ ដល់ ១៨)

លំហាត់ ១៧ Quick Sort (តម្រៀបទំហំអាវយឺត S, M, L, XL - Pythonic Way)

កូដសម្រាយងាយស្រួលយល់ដោយប្រើប្រាស់ List Comprehension៖

Pythonic Quick Sort
def quick_sort(arr):
    if len(arr) <= 1: 
        return arr
    
    # រើសយកធាតុចំកណ្តាលធ្វើជា Pivot (ជួយកាត់បន្ថយ Worst Case លើ Sorted Array)
    pivot = arr[len(arr) // 2]
    
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    
    return quick_sort(left) + middle + quick_sort(right)

sizes_val = [3, 1, 4, 2]  # លេខតំណាងទំហំអាវ៖ S=1, M=2, L=3, XL=4
result = quick_sort(sizes_val)
print("ទំហំអាវតម្រៀបរួច:", result) # Expected Output: [1, 2, 3, 4]

លំហាត់ ១៨ មូលហេតុដែលនាំឱ្យ Quick Sort ពេញនិយមក្នុងឧស្សាហកម្មជាក់ស្តែង

១. In-place Sorting៖ Quick Sort កំណែស្តង់ដារ (ដូចជា Hoare ឬ Lomuto partition) ដំណើរការតម្រៀបដូរទីតាំងលើ Array ដើមផ្ទាល់ នាំឱ្យប្រើប្រាស់ទំហំផ្ទុកបន្ថែមតិចតួចបំផុត $O(\log n)$ ធៀបនឹង Merge Sort ដែលត្រូវការ $O(n)$។
