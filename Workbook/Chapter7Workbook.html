def linear_search(arr, target):
    for i in range(len(arr)):
        if arr[i] == target:
            return i  # រកឃើញ បោះទីតាំង Index ត្រឡប់
    return -1 # រកមិនឃើញ

ស្រាល ២ ស្វែងរកដោយពុះកណ្តាល (Binary Search Iterative)

Copy & Run Code
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target: return mid
        elif arr[mid] < target: left = mid + 1
        else: right = mid - 1
    return -1

# ទិន្នន័យត្រូវតែ Sorted រួចរាល់!
sorted_arr = [10, 20, 30, 40, 50, 70, 80, 90]
print("ទីតាំងលេខ 80 គឺ Index ទី:", binary_search(sorted_arr, 80))

def bs_recursive(arr, target, left, right):
    if left > right: return -1
    
    mid = (left + right) // 2
    if arr[mid] == target:
        return mid
    elif arr[mid] < target:
        return bs_recursive(arr, target, mid + 1, right)
    else:
        return bs_recursive(arr, target, left, mid - 1)

sorted_arr = [10, 20, 30, 40, 50, 70, 80, 90]
print("រក 40 បែប Recursive:", bs_recursive(sorted_arr, 40, 0, len(sorted_arr)-1))

def search_insert(nums, target):
    left, right = 0, len(nums) - 1
    while left <= right:
        mid = (left + right) // 2
        if nums[mid] == target: return mid
        elif nums[mid] < target: left = mid + 1
        else: right = mid - 1
    # បើរកមិនឃើញ left គឺកន្លែងដែលត្រូវស៊កបញ្ចូល

Copy & Run Code
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n - i - 1):
            # ប្រៀបធៀបលេខជាប់គ្នា បើអ្នកមុខធំជាង ដូរទីតាំង!
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr

print("Bubble តម្រៀបរួច:", bubble_sort([64, 34, 25, 12, 22, 11, 90]))
ទោះជាវាស្រួលសរសេរ ប៉ុន្តែវាយឺតខ្លាំងណាស់ O(n²) ព្រោះវាត្រូវដើរប្រៀបធៀបគូលេខម្តងមួយៗច្រើនជុំ ទម្រាំតែអាចរុញលេខធំឱ្យលិចទៅដល់ចុងក្រោយ។

def optimized_bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        swapped = False # ប្រើដើម្បីតាមដាន
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        
        # បើមួយជុំហ្នឹងអត់មានដូរទីតាំងសោះ បានន័យថា Sorted ហើយ!
        if not swapped:
            break
    return arr
បើយើងសង្កេតឃើញថាទិន្នន័យបានរៀបរយស្អាតហើយ យើងឈប់រត់ Loop បន្តទៀតដែលនាំឱ្យខាតពេល (Break loop)។ ធ្វើបែបនេះវាជួយឱ្យ Best Case ក្លាយជា O(n) លឿនជាងមុនឆ្ងាយ។

def insertion_sort(arr):
    # ចាប់ផ្តើមពីសន្លឹកទី ២ ឡើងទៅ
    for i in range(1, len(arr)):
        key = arr[i] # សន្លឹកបៀរដែលកាន់ក្នុងដៃ
        j = i - 1
        
        # រុញអ្នកដែលធំជាង key ទៅមុខមួយកន្លែង
        while j >= 0 and key < arr[j]:
            arr[j + 1] = arr[j]
            j -= 1
            
        # ស៊កបញ្ចូលនៅកន្លែងទំនេរដែលបានរៀបចំ
        arr[j + 1] = key
    return arr

print("Insertion តម្រៀបរួច:", insertion_sort([12, 11, 13, 5, 6]))
វាមានដំណើរការដូចពេលយើងចាប់បៀរហើយស៊កចូលឱ្យត្រូវលេខរៀង។ បើទិន្នន័យស្ទើរតែ Sorted ហើយ វានឹងរត់លឿន O(n) តែបើ Unsorted ខ្លាំងវាយឺត O(n²)។

def insertion_sort_desc(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        # ដូរត្រង់នេះ (key > arr[j]) ដើម្បីតម្រៀបពីធំទៅតូចវិញ
        while j >= 0 and key > arr[j]:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr

print("ធំទៅតូច (Descending):", insertion_sort_desc([12, 11, 13, 5, 6]))

def merge_sort_split(arr):
    if len(arr) <= 1: return arr # Base case អត់អាចពុះទៀតបានទេ
    
    mid = len(arr) // 2 # រកចំណុចកណ្តាល
    left_half = arr[:mid] # ពុះខាងឆ្វេង
    right_half = arr[mid:] # ពុះខាងស្តាំ
    
    print(f"ពុះ: {left_half} និង {right_half}")
    
    # ហៅខ្លួនឯងពុះរហូតដល់សល់ 1
    merge_sort_split(left_half)
    merge_sort_split(right_half)

def merge_two_sorted(left, right):
    merged = []
    i = j = 0
    # ប្រៀបធៀបយកអ្នកតូចដាក់ចូល Array ថ្មីមុនគេ
    while i < len(left) and j < len(right):
        if left[i] < right[j]:
            merged.append(left[i]); i += 1
        else:
            merged.append(right[j]); j += 1
            
    # កើបយកអ្នកសេសសល់ (បើមាន)
    merged.extend(left[i:])
    merged.extend(right[j:])
    return merged

print("ផ្គុំរួច:", merge_two_sorted([3, 27], [9, 82]))
នេះជាជំហានទី២។ វាមាននាទីយក Array តូចៗពីរដែល Sorted រួចរាល់ មកប្រៀបធៀបគ្នាដើម្បីផ្គុំចូលជា Array ធំមួយ (ប្រើពេល O(n)) ដោយប្រើប្រាស់ Space ថ្មី។

def merge_sort(arr):
    if len(arr) <= 1: return arr
    mid = len(arr) // 2
    # ពុះ
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    # ផ្គុំ
    return merge_two_sorted(left, right) # ប្រើ Function ខាងលើ

៤. ករណីសិក្សា៖ ក្បួនតម្រៀប Quick Sort O(n log n)

មធ្យម ៧ Quick Sort កំណែ Pythonic (ងាយស្រួលយល់)

Copy & Run Code
def quick_sort(arr):
    if len(arr) <= 1: return arr # Base case
        
    pivot = arr[len(arr) // 2] # រើសអ្នកកណ្តាលជាមេក្រុម (Pivot)
    
    # បំបែកជា ៣ ក្រុម (ឆ្វេង, កណ្តាល, ស្តាំ) ដោយប្រើ List Comprehension
    left = [x for x in arr if x < pivot]
    mid = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    
    # ហៅខ្លួនឯងឡើងវិញ រួចផ្គុំត្រឡប់
    return quick_sort(left) + mid + quick_sort(right)

print("Quick Sort (Pythonic):", quick_sort([10, 7, 8, 9, 1, 5]))
កូដនេះងាយស្រួលអាន និងងាយយល់បំផុតពីយន្តការរបស់ Quick Sort។ វារើសមេក្រុម រួចបំបែកអ្នកទាបជាងទៅឆ្វេង អ្នកខ្ពស់ជាងទៅស្តាំ។ ប៉ុន្តែវាប្រើប្រាស់ Space O(n) ដើម្បីបង្កើត Array ថ្មី។

ធ្ងន់ ៤ ការរើស Pivot ខុសប្រក្រតី (Worst Case Simulation)

Copy & Run Code
def quick_sort_bad(arr):
    if len(arr) <= 1: return arr
    
    # រើសមេក្រុមយកអ្នកចុងក្រោយគេបង្អស់ជានិច្ច
    pivot = arr[-1] 
    
    left = [x for x in arr[:-1] if x <= pivot]
    right = [x for x in arr[:-1] if x > pivot]
    
    return quick_sort_bad(left) + [pivot] + quick_sort_bad(right)

# បើ Array បាន Sorted រួចហើយ វាជាសត្រូវធំបំផុតរបស់ Quick Sort បែបនេះ
print("តម្រៀប:", quick_sort_bad([1, 2, 3, 4, 5]))

ធ្ងន់ ៥ Quick Sort កំណែស្តង់ដារ (Partition - In-place Swap)

Copy & Run Code
def partition(arr, low, high):
    pivot = arr[high] # យកអ្នកចុងក្រោយជា Pivot
    i = low - 1 # Point ទៅអ្នកដែលតូចជាង Pivot
    
    for j in range(low, high):
        if arr[j] <= pivot:
            i += 1
            arr[i], arr[j] = arr[j], arr[i] # ដូរទីតាំង
            
    arr[i + 1], arr[high] = arr[high], arr[i + 1] # យក Pivot មកដាក់កណ្តាល
    return i + 1 # បោះទីតាំង Pivot ត្រឡប់
នេះគឺជាបេះដូងនៃ Quick Sort ពិតប្រាកដ។ វាប្រើប្រាស់ Pointer ២ (i និង j) ដើម្បីដើរដូរទីតាំងលេខទៅវិញទៅមក (Swap in-place) នៅក្នុង Array ចាស់តែម្តង ដោយមិនបង្កើត Array ថ្មីឡើយ។

ធ្ងន់ ៦ ការតភ្ជាប់ Quick Sort ពេញលេញ (In-place)

Copy & Run Code
def quick_sort_inplace(arr, low, high):
    if low < high:
        # រកទីតាំង Pivot បន្ទាប់ពីបានរៀបឆ្វេង-ស្តាំរួច
        pi = partition(arr, low, high) 
        
        # ហៅខ្លួនឯងតម្រៀបខាងឆ្វេង Pivot 
        quick_sort_inplace(arr, low, pi - 1)
        
        # ហៅខ្លួនឯងតម្រៀបខាងស្តាំ Pivot 
        quick_sort_inplace(arr, pi + 1, high)

arr_q = [10, 7, 8, 9, 1, 5]
quick_sort_inplace(arr_q, 0, len(arr_q)-1)
print("Quick Sort ពិតប្រាកដ:", arr_q)

def majority_element(nums):
    # តម្រៀបលេខពីតូចទៅធំ O(n log n)
    nums.sort() 
    # អ្នកដែលមានចំនួន > n/2 ច្បាស់ជាឈរនៅចំកណ្តាលជានិច្ច!
    return nums[len(nums) // 2]

print("លេខមានច្រើនជាងគេគឺ:", majority_element([2, 2, 1, 1, 1, 2, 2]))

import bisect

def search_range(nums, target):
    # ប្រើប្រាស់ Python built-in Binary Search O(log n)
    left_idx = bisect.bisect_left(nums, target)
    right_idx = bisect.bisect_right(nums, target) - 1
    
    if left_idx <= right_idx and right_idx < len(nums) and nums[left_idx] == target:
        return [left_idx, right_idx]
    return [-1, -1]

print("ចន្លោះលេខ 8:", search_range([5, 7, 7, 8, 8, 10], 8))
Python មាន Library bisect សម្រាប់ធ្វើ Binary Search ស្រាប់ៗ។ bisect_left រកទីតាំងខាងឆ្វេងបំផុត ឯ bisect_right រកទីតាំងបន្ទាប់ពីស្តាំបំផុត។

def missing_number(nums):
    n = len(nums)
    # ប្រើរូបមន្ត Gauss ដើម្បីរកផលបូកសរុប O(1)
    expected_sum = n * (n + 1) // 2
    
    # ផលបូកជាក់ស្តែង O(n)
    actual_sum = sum(nums) 
    
    return expected_sum - actual_sum

print("លេខដែលបាត់គឺ:", missing_number([3, 0, 1])) # គួរតែមាន 2
ជំនួសឱ្យការ Sorting ឬ Linear Search ដើររកលេខបាត់ ដែលស៊ីពេល O(n log n) យឺត យើងបង្វែរមកប្រើរូបមន្តគណិតវិទ្យាបូកដកគ្នា ធ្វើឱ្យល្បឿនស្ទុះដល់ O(n) លឿនបំផុត។

ធ្ងន់ ៨ ការផ្គុំចន្លោះពេលដែលជាន់គ្នា (Merge Intervals)

def merge_intervals(intervals):
    if not intervals: return []
    
    # 1. តម្រៀបតាមលេខចាប់ផ្តើមពីតូចទៅធំ O(n log n)
    intervals.sort(key=lambda x: x[0])
    
    merged = [intervals[0]]
    for current in intervals[1:]:
        last_merged = merged[-1]
        
        # បើជាន់គ្នា (ឧ. [1,3] និង [2,6]) នោះផ្គុំគ្នាជា [1,6]
        if current[0] <= last_merged[1]:
            last_merged[1] = max(last_merged[1], current[1])
        else:
            merged.append(current)
            
    return merged

print("ចន្លោះម៉ោងបញ្ចូលគ្នា:", merge_intervals([[1,3],[2,6],[8,10]]))
