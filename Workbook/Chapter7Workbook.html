Iterative Binary Search
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target: 
            return mid
        elif arr[mid] < target: 
            left = mid + 1
        else: 
            right = mid - 1
    return -1

words = ["Apple", "Banana", "Cherry", "Durian", "Grape", "Mango"]

print("ទីតាំង Durian គឺ Index ទី:", result) # Expected Output: 3

def search_insert_position(grades, new_grade):
    left, right = 0, len(grades) - 1
    while left <= right:
        mid = (left + right) // 2
        if grades[mid] == new_grade: 
            return mid
        elif grades[mid] < new_grade: 
            left = mid + 1
        else: 
            right = mid - 1
    return left  # left ផ្ទុក index សមស្របសម្រាប់ស៊កបញ្ចូល

លំហាត់ ៦ Recursive Binary Search (ក្បួនស្វែងរកពុះកណ្តាលបែប Recursive)

Recursive Binary Search
def binary_search_rec(arr, target, left, right):
    if left > right: 
        return -1
    
    mid = (left + right) // 2
    if arr[mid] == target: 
        return mid
    elif arr[mid] < target:
        return binary_search_rec(arr, target, mid + 1, right)
    else:
        return binary_search_rec(arr, target, left, mid - 1)

numbers = [10, 20, 30, 40, 50, 60]
result = binary_search_rec(numbers, 50, 0, len(numbers) - 1)
print("ស្វែងរកលេខ 50 ឃើញនៅ Index ទី:", result) # Expected Output: 4

លំហាត់ ៧ Selection Sort (តម្រៀបទំហំឯកសាររូបថតពីតូចទៅធំ)

Selection Sort Code
def selection_sort(arr):
    n = len(arr)
    for i in range(n):
        min_idx = i
        for j in range(i + 1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j
        # Swap elements
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
    return arr

file_sizes = [45, 12, 89, 23, 11]
result = selection_sort(file_sizes)
print("ទំហំឯកសារតម្រៀបរួច:", result) # Expected Output: [11, 12, 23, 45, 89]

លំហាត់ ៩ Optimized Bubble Sort (តម្រៀបកម្ពស់សិស្សគោរពទង់ជាតិ)

Optimized Bubble Sort
def bubble_sort_optimized(arr):
    n = len(arr)
    for i in range(n):
        swapped = False
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        if not swapped: 
            break  # បញ្ឈប់ការរត់ loop ប្រសិនបើគ្មានការ Swap សោះ
    return arr

លំហាត់ ១១ Insertion Sort (ការតម្រៀបសន្លឹកបៀរលេងនៅលើដៃ)

Insertion Sort Code
def insertion_sort(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        # រុញធាតុដែលធំជាង key ទៅខាងស្តាំដើម្បីបើកផ្លូវ
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr

cards = [10, 2, 5, 3, 9]
result = insertion_sort(cards)
print("សន្លឹកបៀរតម្រៀបរួច:", result) # Expected Output: [2, 3, 5, 9, 10]

def insertion_sort_steps(arr):
    steps = 0
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        while j >= 0 and arr[j] > key:
            steps += 1 # រាប់ពេលមានការរុញ (shift)
            arr[j + 1] = arr[j]
            j -= 1
        steps += 1 
        arr[j + 1] = key
    return steps

print("Insertion Sort ត្រូវការប្រៀបធៀបសរុប:", insertion_sort_steps(nearly_sorted), "ដង")

លំហាត់ ១៤ Merge Arrays (ដំណើរការច្របាច់បញ្ចូល list ដែល Sorted ពីរចូលគ្នា)

Merge Arrays Helper Function
def merge_arrays(left, right):
    res = []
    i = j = 0
    while i < len(left) and j < len(right):
        if left[i] < right[j]:
            res.append(left[i])
            i += 1
        else:
            res.append(right[j])
            j += 1
            
    res.extend(left[i:])
    res.extend(right[j:])
    return res

result = merge_arrays([101, 105], [102, 103])
print("ច្របាច់បញ្ចូលរួច:", result)

def merge_sort(arr):

លំហាត់ ១៧ Quick Sort (តម្រៀបទំហំអាវ S, M, L, XL - Pythonic Way)

Pythonic Quick Sort
def quick_sort(arr):
    if len(arr) <= 1: 
        return arr
    
    pivot = arr[len(arr) // 2]
    
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    
    return quick_sort(left) + middle + quick_sort(right)

sizes_val = [3, 1, 4, 2]
result = quick_sort(sizes_val)
print("ទំហំអាវតម្រៀបរួច:", result)

លំហាត់ ១៨ In-Place Quick Sort (Lomuto Partition Scheme)

In-Place Quick Sort
def partition(arr, low, high):
    pivot = arr[high]
    i = low - 1
    for j in range(low, high):
        if arr[j] < pivot:
            i += 1
            arr[i], arr[j] = arr[j], arr[i] # In-place swap (មិនស៊ីទំហំថ្មី)
    arr[i + 1], arr[high] = arr[high], arr[i + 1]
    return i + 1

def quick_sort_inplace(arr, low, high):
    if low < high:
        pi = partition(arr, low, high)
        quick_sort_inplace(arr, low, pi - 1)
        quick_sort_inplace(arr, pi + 1, high)

arr = [10, 7, 8, 9, 1, 5]
quick_sort_inplace(arr, 0, len(arr)-1)
print("តម្រៀបដោយមិនប្រើ Array ថ្មី:", arr)

លំហាត់ ២១ Interpolation Search (ការស្វែងរកបែបប៉ាន់ស្មានលើទិន្នន័យដែលចែករំលែកស្មើគ្នា)

Interpolation Search Algorithm
def interpolation_search(arr, target):
    low, high = 0, len(arr) - 1
    while low <= high and target >= arr[low] and target <= arr[high]:
        if low == high:
            if arr[low] == target: return low
            return -1
        
        pos = low + int(((high - low) / (arr[high] - arr[low])) * (target - arr[low]))
        
        if arr[pos] == target: return pos
        if arr[pos] < target: low = pos + 1
        else: high = pos - 1
    return -1

uniform_data = [10, 20, 30, 40, 50, 60, 70, 80, 90]
print("ទីតាំងលេខ 70 គឺ Index ទី:", interpolation_search(uniform_data, 70))

ល្បឿនមធ្យមរបស់ Interpolation Search គឺលឿនរហូតដល់ O(log(log n)) ដែលលឿនជាង Binary Search (O(log n)) នៅលើទិន្នន័យដែលចែកចែកចាយស្មើគ្នា (Uniform Distribution) ព្រោះវាគណនាស្វែងរកចំគោលដៅប្រហាក់ប្រហែល។

លំហាត់ ២២ Searching in 2D Matrix (ការស្វែងរកតម្លៃក្នុង matrix ២ វិមាត្រដែលបានតម្រៀបរួច)

2D Matrix Binary Search
def search_2d_matrix(matrix, target):
    if not matrix or not matrix[0]: return False
    rows, cols = len(matrix), len(matrix[0])
    low, high = 0, rows * cols - 1
    
    while low <= high:
        mid = (low + high) // 2
        mid_val = matrix[mid // cols][mid % cols]
        
        if mid_val == target: return True
        elif mid_val < target: low = mid + 1
        else: high = mid - 1
    return False

matrix = [[1, 3, 5], [10, 11, 16], [23, 30, 34]]
print("តើមានលេខ 16 ដែរឬទេ?:", search_2d_matrix(matrix, 16))

ក្បួនតម្រៀបត្រូវបានចាត់ទុកថាមាន ស្ថេរភាព (Stable) កាលណាវាធានារក្សាទីតាំងដើមនៃធាតុទាំងឡាយណាដែលមានគន្លឹះ (key) ស្មើគ្នា។
- Stable Sorts: Bubble Sort, Insertion Sort, Merge Sort, Timsort។
- Unstable Sorts: Selection Sort, Quick Sort, Heap Sort។

  • ១. ប្រើប្រាស់ Built-in Functions (Timsort): ក្នុងការអភិវឌ្ឍន៍កម្មវិធីទូទៅ សូមប្រើប្រាស់ .sort() ឬ sorted() របស់ Python ជានិច្ច ព្រោះវាជាប្រភេទក្បួនកាត់កូនកាត់កម្រិតខ្ពស់ដែលមានប្រសិទ្ធភាពខ្ពស់ និងរត់បានលឿនបំផុតលើទិន្នន័យពិភពលោកពិត (O(n log n))។
