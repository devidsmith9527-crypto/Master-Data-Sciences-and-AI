ធ្ងន់ ២ បច្ចេកទេសបង្វិល Array (Two-Pointers Reverse)

Copy & Run Code
def reverse_in_place(arr):
    left, right = 0, len(arr) - 1
    while left < right:
        # ប្តូរទីតាំងគ្នា (Swap values)
        arr[left], arr[right] = arr[right], arr[left]
        left += 1
        right -= 1
    return arr

print("បញ្ចីក្រោយបង្វិល:", reverse_in_place([1, 2, 3, 4]))
បច្ចេកទេសនេះចំណេញ Memory (Space O(1)) ដោយសារយើងគ្រាន់តែយកតម្លៃខាងឆ្វេងមកឆ្លាស់ជាមួយតម្លៃខាងស្តាំដោយផ្ទាល់ មិនចាំបាច់បង្កើត List ថ្មីមួយទៀតឡើយ។

# Pop (ដកទិន្នន័យចេញពីលើគេ)

print("យកចេញនូវ:", top_item)

ធ្ងន់ ១ ការប្រើ Stack បញ្ច្រាសអក្សរ (String Reversal)

Copy & Run Code
def reverse_string(text):
    stack = []
    # Push អក្សរចូលម្តងមួយៗ
    for char in text:
        stack.append(char)
        
    result = ""
    # Pop ចេញមកវិញ (LIFO)
    while stack:
        result += stack.pop()
    return result

print("ពាក្យ HELLO បញ្ច្រាសទៅជា:", reverse_string("HELLO"))
ដោយសារ Stack ប្រើច្បាប់ LIFO ពេលយើង Push 'H', 'E', 'L', 'L', 'O' ចូល នោះអក្សរ 'O' ស្ថិតនៅលើគេ។ ពេល Pop មកវិញ វានឹងចេញអក្សរ 'O' មុនគេ រួចបន្តដល់ 'H' ទើបចេញជា 'OLLEH'។

៤. ករណីសិក្សា៖ ការដោះស្រាយបញ្ហាដោយប្រើ Stack

ធ្ងន់ ១ ការប្រើ Stack ឆែកវង់ក្រចក (Balanced Parentheses)

Copy & Run Code
def is_balanced(expr):
    stack = []
    for char in expr:
        if char in "({[": 
            stack.append(char)
        elif char in ")}]":
            if not stack: return False
            stack.pop() # ដកអាបើកចុងក្រោយចេញមកផ្ទឹម
    return len(stack) == 0

print("កូដ (a+b)*[c] ត្រូវទេ?", is_balanced("(a+b)*[c]"))
print("កូដ (a+b] ត្រូវទេ?", is_balanced("(a+b]"))
Stack ស័ក្តិសមបំផុតសម្រាប់ដោះស្រាយបញ្ហានេះ។ រាល់ពេលជួបសញ្ញាបើក វា Push ចូល។ ពេលជួបសញ្ញាបិទ វា Pop ចេញ។ បើកូដត្រឹមត្រូវ ចុងក្រោយ Stack ត្រូវតែទទេស្អាត។

undo_stack = []

# អ្នកប្រើប្រាស់វាយអត្ថបទ

print("សកម្មភាពនៅសល់:", undo_stack)

៥. ករណីសិក្សា៖ រចនាសម្ព័ន្ធ Queue (FIFO)

ស្រាល ១ បង្កើត Queue ដោយប្រើ List ធម្មតា (បញ្ហាយឺត)

Copy & Run Code
queue_list = []

# ចូលជួរពីខាងក្រោយ
queue_list.append("អតិថិជន ១") 
queue_list.append("អតិថិជន ២")

# ចេញពីជួរ (បម្រើអ្នកមកមុនគេ ដោយដកពី Index 0)
first = queue_list.pop(0) 

print("បានបម្រើ:", first)
ទោះបីកូដនេះដើរត្រូវតាមច្បាប់ FIFO តែការប្រើ pop(0) លើ List ធម្មតា គឺយឺតខ្លាំងកម្រិត O(n) ព្រោះកុំព្យូទ័រត្រូវរុញទិន្នន័យក្រោយៗទាំងអស់មកមុខមួយកៅអីម្នាក់ជានិច្ច។

មធ្យម ១ ការប្រើប្រាស់ deque សម្រាប់ Queue លឿនបំផុត

Copy & Run Code
from collections import deque

fast_q = deque()

# Enqueue (ចូលជួរ)
fast_q.append("Customer A")
fast_q.append("Customer B")

# Dequeue ពីខាងមុខដោយល្បឿន O(1)
served = fast_q.popleft() 

print("Served:", served)
print("Remaining in Queue:", fast_q)
deque (Double-Ended Queue) ជា Data Structure ដែលបង្កើតមកជាពិសេសអនុញ្ញាតឱ្យយើងបន្ថែម និងដកទិន្នន័យពីខាងមុខ (popleft()) បានយ៉ាងរហ័ស ក្នុងល្បឿនថេរ O(1)។

៦. ករណីសិក្សា៖ ការដោះស្រាយបញ្ហាដោយប្រើ Queue

មធ្យម ១ ការប្រើ Queue បង្កើតម៉ាស៊ីនព្រីន (Printer Queue)

Copy & Run Code
from collections import deque

print_jobs = deque(["Doc_Math.pdf", "Doc_IT.pdf", "Doc_History.pdf"])

while print_jobs:
    # ទាញយកឯកសារទី១ ចេញមកព្រីន
    job = print_jobs.popleft()
    print(f"កំពុងបោះពុម្ព: {job}...")
    
print("បញ្ចប់កិច្ចការព្រីន!")
Loop នឹងដំណើរការដកឯកសារចេញពីជួរមុខម្តងមួយៗ (FIFO) មករួចព្រីន ទាល់តែអស់ឯកសារពីក្នុង print_jobs ទើបវាលោតចេញមកក្រៅ និងបង្ហាញពាក្យ "បញ្ចប់កិច្ចការព្រីន!"។

មធ្យម ២ ការបង្កើតសេវាកម្មធនាគារ (Bank Teller)

Copy & Run Code
from collections import deque
import time

bank_queue = deque(["Tola", "Bopha"])

def serve_customers():
    while bank_queue:
        customer = bank_queue.popleft()
        print(f"កំពុងបម្រើសេវាជូន: {customer}")
        time.sleep(1) # ចំណាយពេល ១ វិនាទី

serve_customers()

៧. ករណីសិក្សា៖ Queue កម្រិតខ្ពស់ និង AI Applications

ធ្ងន់ ១ ការបង្វិលទិន្នន័យក្នុងជួរ Queue (Rotate Queue)

Copy & Run Code
from collections import deque

q = deque([1, 2, 3, 4, 5])

# រុញធាតុ 2 គ្រាប់ខាងមុខ ទៅដាក់កន្ទុយក្រោយវិញ
q.rotate(-2) 

print("បន្ទាប់ពី Rotate:", q)
លទ្ធផលគឺ deque([3, 4, 5, 1, 2])។ អនុគមន៍ rotate(-2) បានរុញទិន្នន័យ ២ គ្រាប់ខាងមុខ (លេខ 1 និង 2) អោយទៅនៅចុងជួរខាងក្រោយវិញយ៉ាងងាយស្រួលដោយមិនបាច់សរសេរ Loop ដកដាក់ៗនោះទេ។

មធ្យម ២ ការប្រើប្រាស់ Queue ក្នុងក្បួន AI (BFS Concept)

Copy & Run Code
from collections import deque

# ក្រាហ្វតំណាងទីក្រុងដែលភ្ជាប់គ្នា
graph = {'A': ['B', 'C'], 'B': ['D'], 'C': [], 'D': []}

def bfs(start):
    queue = deque([start])
    while queue:
        node = queue.popleft()
        print(node, end=" ")
        for neighbor in graph[node]:
            queue.append(neighbor)

print("លទ្ធផល BFS ស្វែងរកផ្លូវ:"); bfs('A')

៨. ករណីសិក្សា៖ ប្រតិបត្តិការចម្រុះ និង Time Complexity

ធ្ងន់ ១ បង្កើត Queue មួយ ដោយប្រើប្រាស់ Stacks ពីរ

Copy & Run Code
class MyQueue:
    def __init__(self):
        self.s1 = []; self.s2 = []
    def enqueue(self, x):
        self.s1.append(x)
    def dequeue(self):
        if not self.s2:
            while self.s1:
                self.s2.append(self.s1.pop())
        return self.s2.pop()

q = MyQueue()
q.enqueue(10); q.enqueue(20); q.enqueue(30)
print("ទាញចេញមុនគេគឺ:", q.dequeue()) # លទ្ធផល: 10
នេះជាលំហាត់សម្ភាសន៍ទូទៅ (Interview Question)។ យើងបំប្លែង LIFO ទៅជា FIFO ដោយចាក់ធាតុពី Stack ទី១ ចូលទៅក្នុង Stack ទី២ ធ្វើឱ្យទិន្នន័យនៅក្រោមគេ លោតមកនៅខាងលើគេបង្អស់វិញទើបយើង Pop ចេញ។

temperatures = [35, 38, 32, 40, 36]
