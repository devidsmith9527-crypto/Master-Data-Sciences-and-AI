២. ករណីសិក្សា៖ ប្រតិបត្តិការរាប់ចំនួនក្នុង Tree

មធ្យម ១ ការរាប់ចំនួន Nodes សរុបនៅក្នុង Tree

Copy & Run Code
class TreeNode:
    def __init__(self, val): self.val=val; self.left=self.right=None

def countNodes(root):
    if not root: return 0
    # យក ១ (ខ្លួនឯង) បូកចំនួនកូនឆ្វេង និងកូនស្តាំ
    return 1 + countNodes(root.left) + countNodes(root.right)

r = TreeNode(1); r.left = TreeNode(2); r.right = TreeNode(3)
print("ចំនួន Nodes សរុប:", countNodes(r))
អនុគមន៍នេះវាចុះទៅរាប់គ្រប់មែកទាំងអស់រហូតដល់ចុងបញ្ចប់ (O(n)) ដោយយក ១ តំណាងឱ្យមេម្នាក់ៗបូកបញ្ចូលគ្នាដើម្បីបានចំនួនសរុប។

មធ្យម ២ ការរាប់ចំនួន Leaf Nodes ទាំងអស់ (Nodes គ្មានកូន)

Copy & Run Code
class TreeNode:
    def __init__(self, val): self.val=val; self.left=self.right=None

def countLeaves(root):
    if not root: return 0
    # បើគ្មានកូនឆ្វេង និងកូនស្តាំ គឺវាជា Leaf (ស្លឹក)
    if not root.left and not root.right: return 1
    return countLeaves(root.left) + countLeaves(root.right)

r = TreeNode(1); r.left = TreeNode(2); r.right = TreeNode(3)
r.left.left = TreeNode(4) # បន្ថែមកូនឲ្យលេខ 2
print("ចំនួនស្លឹក (Leaves) សរុប:", countLeaves(r))

៣. ករណីសិក្សា៖ ការវាស់វែងកម្ពស់ និងផលបូក

ធ្ងន់ ១ ការគណនាកម្ពស់ (Max Depth/Height) នៃ Tree

Copy & Run Code
class TreeNode:
    def __init__(self, val): self.val=val; self.left=self.right=None

def maxDepth(root):
    if not root: return 0
    return 1 + max(maxDepth(root.left), maxDepth(root.right))

r = TreeNode(1); r.left = TreeNode(2); r.left.left = TreeNode(4)
print("កម្ពស់អតិបរមារបស់ Tree គឺ:", maxDepth(r))

class TreeNode:
    def __init__(self, val): self.val=val; self.left=self.right=None

def sumTree(root):
    if not root: return 0
    # យកតម្លៃខ្លួនឯង បូកនឹងកូនចៅឆ្វេង និងស្តាំទាំងអស់
    return root.val + sumTree(root.left) + sumTree(root.right)

៤. ករណីសិក្សា៖ Binary Search Tree (BST) ការបញ្ចូលទិន្នន័យ

ស្រាល ៤ ការបញ្ចូលទិន្នន័យទៅក្នុង BST (Recursive Insert)

Copy & Run Code
class TreeNode:
    def __init__(self, val): self.val=val; self.left=self.right=None

def insert(root, key):
    if not root: return TreeNode(key)
    if key < root.val: root.left = insert(root.left, key)
    elif key > root.val: root.right = insert(root.right, key)
    return root

r = TreeNode(50)
r = insert(r, 30); r = insert(r, 70)
print("កូនឆ្វេង:", r.left.val, " | កូនស្តាំ:", r.right.val)

class TreeNode:
    def __init__(self, val): self.val=val; self.left=self.right=None

៥. ករណីសិក្សា៖ ការបញ្ចូល Iterative និងការត្រួតពិនិត្យ BST

មធ្យម ៤ ការបញ្ចូលទិន្នន័យដោយមិនប្រើ Recursive (Iterative Insert)

Copy & Run Code
class TreeNode:
    def __init__(self, val): self.val=val; self.left=self.right=None

def insertIter(root, key):
    if not root: return TreeNode(key)
    curr = root
    while True:
        if key < curr.val:
            if not curr.left: curr.left = TreeNode(key); break
            curr = curr.left
        else:
            if not curr.right: curr.right = TreeNode(key); break
            curr = curr.right
    return root

r = insertIter(None, 10); insertIter(r, 5)
print("កូនឆ្វេង:", r.left.val)
កូដនេះប្រើ While Loop ដើររកកន្លែងទំនេរដោយផ្ទាល់។ វាជួយសន្សំសំចៃ Memory (Space O(1)) ព្រោះវាមិនមានផ្ទុក Call Stack រាប់លានដងដូចវិធី Recursive ឡើយ។

ធ្ងន់ ២ ការត្រួតពិនិត្យថាវាជា BST ត្រឹមត្រូវឬទេ (Validate BST)

Copy & Run Code
class TreeNode:
    def __init__(self, val): self.val=val; self.left=self.right=None

def isValidBST(root, min_val=float('-inf'), max_val=float('inf')):
    if not root: return True
    if not (min_val < root.val < max_val): return False
    
    # ឆែកកូនឆ្វេង (ត្រូវតូចជាង Root) និង កូនស្តាំ (ត្រូវធំជាង Root)
    return isValidBST(root.left, min_val, root.val) and \
           isValidBST(root.right, root.val, max_val)

r = TreeNode(2); r.left = TreeNode(1); r.right = TreeNode(3)
print("តើ Tree នេះជា BST ត្រឹមត្រូវទេ?", isValidBST(r))

៦. ករណីសិក្សា៖ ការស្វែងរកទិន្នន័យក្នុង BST (Search)

ស្រាល ៥ ស្វែងរកលេខក្នុង BST (Recursive Search)

Copy & Run Code
class TreeNode:
    def __init__(self, val): self.val=val; self.left=self.right=None

def search(root, key):
    if not root or root.val == key: return root
    
    # បើលេខដែលចង់រកតូចជាង ត្រូវទៅឆ្វេង ឯលេខធំជាងទៅស្តាំ
    if key < root.val: return search(root.left, key)
    return search(root.right, key)

r = TreeNode(50); r.left = TreeNode(30); r.right = TreeNode(70)
result = search(r, 30)

    return curr

    curr = root
    while curr:
        if abs(curr.val - target) < abs(closest - target):
            closest = curr.val
        curr = curr.left if target < curr.val else curr.right
    return closest

def minValueNode(node):
    current = node
    while current.left is not None:
        current = current.left
    return current

def deleteNode(root, key):
    if not root: return root
    if key < root.val: root.left = deleteNode(root.left, key)
    elif key > root.val: root.right = deleteNode(root.right, key)
    else:
        if root.left is None: return root.right
        elif root.right is None: return root.left
        temp = minValueNode(root.right)
        root.val = temp.val
        root.right = deleteNode(root.right, temp.val)
    return root

class TreeNode:
    def __init__(self, val): self.val=val; self.left=self.right=None

def kthSmallest(root, k):
    stack = []
    while True:
        while root:
            stack.append(root)
            root = root.left # ចុះទៅឆ្វេងឲ្យអស់លេខតូចៗ
        
        root = stack.pop()
        k -= 1
        if k == 0: return root.val # រកឃើញអ្នកទី K
        
        root = root.right

r = TreeNode(50); r.left = TreeNode(30); r.right = TreeNode(70)
print("អ្នកតូចជាងគេទី 2 គឺ:", kthSmallest(r, 2))

def evaluateLoan(creditScore, hasCollateral):
    # កម្រិតទី១: ពិនិត្យ Credit Score
    if creditScore < 60:
        return "Reject"  # Leaf Node ១
    else:
        # កម្រិតទី២: ពិនិត្យទ្រព្យបញ្ចាំ
        if hasCollateral:
            return "Approve"  # Leaf Node ២
        else:
            return "Reject"   # Leaf Node ៣
