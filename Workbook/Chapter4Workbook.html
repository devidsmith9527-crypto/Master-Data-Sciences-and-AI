ស្រាល ៣ ការតភ្ជាប់ Nodes ៣ និងតួនាទីរបស់ Head

Copy & Run Code
class Node:
    def __init__(self, d): self.data = d; self.next = None

# បង្កើត Nodes ៣ ដាច់ពីគ្នា

nodeB = Node("B")
nodeC = Node("C")

# តភ្ជាប់ខ្សែបញ្ជូលគ្នា
head.next = nodeB

ចេញលទ្ធផល "C"។ head គឺសំខាន់បំផុត ព្រោះបើបាត់បង់ Head យើងនឹងបាត់បង់ការចង្អុលទៅកាន់ Node បន្តបន្ទាប់ទាំងអស់នៅក្នុងបណ្តាញ។

មធ្យម ១ ការរុករកបង្ហាញទិន្នន័យទាំងអស់ (Traversal)

Copy & Run Code
class Node:
    def __init__(self, d): self.data = d; self.next = None

def print_sll(head):
    curr = head
    while curr is not None:
        print(curr.data, end=" -> ")
        curr = curr.next # រំកិលទៅមុខតាមខ្សែបន្ទាប់
    print("None")

h = Node(1); h.next = Node(2); h.next.next = Node(3)
print_sll(h)
អថេរ curr ដើរតួជាអ្នកដើរតាមខ្សែពីក្បាលរហូតដល់ប្រអប់ចុងក្រោយ (None) រួចផ្តិតយកទិន្នន័យបង្ហាញ។ បើមាន N ប្រអប់ វារត់ N ដង (O(n))។

៣. ករណីសិក្សា៖ ការបន្ថែម និងការរាប់ក្នុង SLL

មធ្យម ២ ការរាប់ចំនួន Nodes សរុប (Length of SLL)

Copy & Run Code
class Node:
    def __init__(self, d): self.data = d; self.next = None

def count_nodes(head):
    count = 0
    curr = head
    while curr:
        count += 1
        curr = curr.next
    return count

head = Node(10); head.next = Node(20)
print("ចំនួន Nodes សរុប:", count_nodes(head))
សម្រាប់ List (Array) ធម្មតា len() ដើរក្នុងល្បឿន O(1) តែសម្រាប់ Linked List យើងត្រូវដើររាប់ម្តងមួយប្រអប់ ទើបវាប្រើពេលរហូតដល់ O(n)។

៤. ករណីសិក្សា៖ ប្រតិបត្តិការចុងបញ្ជី និងការលុប

មធ្យម ៣ បញ្ចូល Node ថ្មីនៅខាងចុង (Insert at Tail)

Copy & Run Code
class Node:
    def __init__(self, d): self.data = d; self.next = None

def insert_tail(head, data):
    new_node = Node(data)
    if not head: return new_node
    curr = head
    while curr.next:  # ដើររហូតដល់កន្ទុយចុងក្រោយគេ
        curr = curr.next
    curr.next = new_node
    return head

h = insert_tail(None, 10); insert_tail(h, 20)
print(h.data, "->", h.next.data)
ដោយសារយើងអត់មានខ្សែចង្អុលទៅកន្ទុយ (Tail Pointer) ផ្ទាល់ យើងត្រូវប្រើ While Loop ដើម្បីដើរពីក្បាលរហូតដល់ប្រអប់ចុងក្រោយ ទើបអាចចងខ្សែបញ្ជូលវាបាន។

៥. ករណីសិក្សា៖ ការស្វែងរក និងលុបតាមតម្លៃក្នុង SLL

ធ្ងន់ ១ ការស្វែងរកទិន្នន័យក្នុងបញ្ជី (Search Value)

Copy & Run Code
class Node:
    def __init__(self, d): self.data = d; self.next = None

def search_node(head, target):
    curr = head
    while curr:
        if curr.data == target:
            return True
        curr = curr.next
    return False

h = Node("X"); h.next = Node("Y")
print("តើមាន Y ទេ?", search_node(h, "Y"))
print("តើមាន Z ទេ?", search_node(h, "Z"))
វាដំណើរការដូច Linear Search ក្នុង Array ដែរ (O(n)) ដោយត្រូវធៀបទិន្នន័យម្តងមួយៗពីក្បាលរហូតដល់ចុងបញ្ជី។

ធ្ងន់ ២ ការលុប Node កណ្តាល ដោយស្វែងរកតម្លៃ (Delete Value)

Copy & Run Code
class Node:
    def __init__(self, d): self.data = d; self.next = None

def delete_value(head, target):
    if not head: return None
    if head.data == target: return head.next # លុបចំ Head
    
    curr = head
    while curr.next and curr.next.data != target:
        curr = curr.next # ដើររកមើលប្រអប់បន្ទាប់
        
    if curr.next: 
        curr.next = curr.next.next # កាត់ខ្សែរំលងប្រអប់គោលដៅ
    return head

h = Node(1); h.next = Node(2); h.next.next = Node(3)
h = delete_value(h, 2)
print("លុបលេខ២ចេញ:", h.data, "->", h.next.data)
យើងត្រូវឈប់នៅប្រអប់ពីមុខគោលដៅ (curr) រួចភ្ជាប់ខ្សែ next របស់វា រំលងទៅប្រអប់បន្ទាប់មួយទៀត នោះប្រអប់កណ្តាលនឹងត្រូវកាត់ផ្តាច់។

៦. ករណីសិក្សា៖ Doubly Linked List (DLL)

ស្រាល ៦ បង្កើត Node សម្រាប់ DLL (មានខ្សែពីរសងខាង)

Copy & Run Code
class DNode:
    def __init__(self, data):
        self.prev = None # ខ្សែត្រឡប់ក្រោយ
        self.data = data
        self.next = None # ខ្សែទៅមុខ

n1 = DNode(100)
print("ប្រអប់ DLL:", n1.data)
DNode មានខ្សែបន្ថែមមួយទៀតគឺ prev សម្រាប់ចង្អុលទៅកាន់ប្រអប់ដែលនៅពីមុខវា ដែលជួយឱ្យវាអាចធ្វើដំណើរថយក្រោយបាន តែវាក៏ស៊ី Memory ជាងមុនដែរ។

ស្រាល ៧ ការតភ្ជាប់ DNodes ចូលគ្នា (Link Forward/Backward)

Copy & Run Code
class DNode:
    def __init__(self, d): self.prev=None; self.data=d; self.next=None

n1 = DNode("Page 1")
n2 = DNode("Page 2")

# ចងខ្សែទាំងសងខាង
n1.next = n2   # n1 ទៅមុខជួប n2
n2.prev = n1   # n2 ថយក្រោយជួប n1

print("ពី Page 2 ថយក្រោយមក:", n2.prev.data)
រាល់ពេលតភ្ជាប់ DNode យើងត្រូវប្រាកដថាបានតភ្ជាប់ខ្សែទាំងសងខាង។ n1.next ចង្អុលទៅ n2 ហើយ n2.prev ត្រូវចង្អុលត្រឡប់មក n1 វិញ។

Copy & Run Code
class DNode:
    def __init__(self, d): self.prev=None; self.data=d; self.next=None

def print_forward(head):
    curr = head
    while curr:
        print(curr.data, end=" -> ")
        curr = curr.next
    print("None")

h = DNode("A"); m = DNode("B"); h.next = m; m.prev = h
print_forward(h)

class DNode:
    def __init__(self, d): self.prev=None; self.data=d; self.next=None

    new_node.next = head
    if head:
        head.prev = new_node # ខ្សែត្រឡប់របស់ Head ចាស់
    return new_node # ក្លាយជា Head ថ្មី

Copy & Run Code
class DNode:
    def __init__(self, d): self.prev=None; self.data=d; self.next=None

def insert_dll_tail(head, data):
    new_node = DNode(data)
    if not head: return new_node
    curr = head
    while curr.next: curr = curr.next # ដើររកកន្ទុយ
    
    curr.next = new_node
    new_node.prev = curr # ភ្ជាប់ត្រឡប់មកវិញ
    return head

h = DNode(10); h = insert_dll_tail(h, 20)

មធ្យម ៦ បំប្លែង Array ទៅជា Linked List (Array to SLL)

Copy & Run Code
class Node:
    def __init__(self, d): self.data=d; self.next=None

def array_to_sll(arr):
    if not arr: return None
    head = Node(arr[0])
    curr = head
    for val in arr[1:]:
        curr.next = Node(val)
        curr = curr.next
    return head

h = array_to_sll([1, 2, 3])
print(h.data, "->", h.next.data, "->", h.next.next.data)
យើងយកធាតុដំបូងរបស់ Array មកធ្វើជា Head រួច Loop យកធាតុបន្ទាប់ៗមកបង្កើតជា Node ថ្មី ហើយតភ្ជាប់ខ្សែ next ជាបន្តបន្ទាប់រហូតដល់ចប់បញ្ជី Array។

class Node:
    def __init__(self, d): self.data=d; self.next=None

def find_middle(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next       # ដើរម្តង ១ ជំហាន
        fast = fast.next.next  # ដើរម្តង ២ ជំហាន
    return slow.data

១០. ករណីសិក្សា៖ ការបំប្លែងទម្រង់ Linked Lists

ធ្ងន់ ៦ ត្រឡប់ខ្សែ Singly Linked List បញ្ច្រាស (Reverse SLL)

Copy & Run Code
class Node:
    def __init__(self, d): self.data=d; self.next=None

def reverse_list(head):
    prev = None; curr = head
    while curr:
        nxt = curr.next  # រក្សាទុកខ្សែទៅមុខសិន
        curr.next = prev # កាត់ខ្សែបង្វិលមកក្រោយវិញ
        prev = curr; curr = nxt # រំកិលទៅមុខបន្ត
    return prev # prev ក្លាយជា Head ថ្មីនៅខាងចុង

h = Node(1); h.next=Node(2)
rev = reverse_list(h)
print("បង្វិលខ្សែ:", rev.data, "->", rev.next.data)
យើងត្រូវការអថេរ ៣ ដើម្បីចងចាំខ្សែអនាគតសិន (nxt) មុននឹងកាត់ខ្សែបច្ចុប្បន្នចង្អុលមកអតីតកាលវិញ (prev)។ វាដំណើរការក្នុងល្បឿន O(n) និងចំណេញ Memory។

មធ្យម ៧ លុបលេខស្ទួនពីបញ្ជី SLL ដែលតម្រៀបរួច

class Node:
    def __init__(self, d): self.data=d; self.next=None

def remove_dups(head):
    curr = head
    while curr and curr.next:
        if curr.data == curr.next.data:
            curr.next = curr.next.next # កាត់ខ្សែរំលងអាស្ទួនចោល
        else: curr = curr.next
    return head

h = Node(1); h.next=Node(1); h.next.next=Node(2)
clean = remove_dups(h)
print("លុបស្ទួន:", clean.data, "->", clean.next.data)

ធ្ងន់ ៧ រាវរកផ្លូវទាល់ ឬរង្វិលជុំ (Detect Cycle)

Copy & Run Code
class Node:
    def __init__(self, d): self.data=d; self.next=None

def has_cycle(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next; fast = fast.next.next
        if slow == fast: return True
    return False

n1=Node(1); n2=Node(2); n3=Node(3)
n1.next=n2; n2.next=n3; n3.next=n1 # ចងខ្សែវល់គ្នា (Cycle)
print("មាន Cycle ទេ?", has_cycle(n1))

ធ្ងន់ ៨ ផ្តុំបញ្ជីតម្រៀបរួចពីរចូលគ្នា (Merge 2 Sorted SLLs)

Copy & Run Code
class Node:
    def __init__(self, d): self.data=d; self.next=None

def merge(l1, l2):
    dummy = curr = Node(0)
    while l1 and l2:
        if l1.data < l2.data: curr.next = l1; l1 = l1.next
        else: curr.next = l2; l2 = l2.next
        curr = curr.next
    curr.next = l1 or l2 
    return dummy.next

list1 = Node(1); list1.next = Node(3)
list2 = Node(2); list2.next = Node(4)
m = merge(list1, list2)
print("បញ្ចួលគ្នា:", m.data, m.next.data, m.next.next.data)

មធ្យម ៨ ការយក SLL មកបង្កើតជា Stack (LIFO)

Copy & Run Code
class Node:
    def __init__(self, d): self.data=d; self.next=None
class StackLL:
    def __init__(self): self.top = None
    def push(self, data):
        n = Node(data)
        n.next = self.top; self.top = n
    def pop(self):
        if not self.top: return None
        val = self.top.data; self.top = self.top.next
        return val

s = StackLL(); s.push(10); s.push(20)
print("យកចេញ (Pop):", s.pop()) # លទ្ធផល: 20
ការប្រើប្រាស់ Linked List បង្កើត Stack មានចំណុចល្អជាង Array ត្រង់ថា វាអាចផ្ទុកទិន្នន័យបានច្រើនតាមតែ Memory អាចទទួលយកបាន (Dynamic Allocation) ដោយល្បឿន O(1) ជានិច្ច។

ធ្ងន់ ៩ ការយក SLL មកបង្កើតជា Queue (FIFO)

Copy & Run Code
class Node:
    def __init__(self, d): self.data=d; self.next=None
class QueueLL:
    def __init__(self): self.head = self.tail = None
    def enqueue(self, d):
        n = Node(d)
        if not self.tail: self.head = self.tail = n; return
        self.tail.next = n; self.tail = n

q = QueueLL(); q.enqueue("A"); q.enqueue("B")
print("អ្នកមកមុនគេគឺ (Head):", q.head.data) # លទ្ធផល: A
