def find_max(arr):
    max_val = arr[0]
    for x in arr:
        if x > max_val:
            max_val = x
    return max_val

Copy & Run Code
def reverse_arr(arr):
    left = 0
    right = len(arr) - 1
    while left < right:
        arr[left], arr[right] = arr[right], arr[left]
        left += 1
        right -= 1
    return arr

print("បង្វិល:", reverse_arr([1, 2, 3, 4, 5]))
ដោយសារ Loop នេះរត់តែពាក់កណ្តាល Array (N/2) ព្រោះវាប្តូរទីតាំងសងខាង។ តែក្នុងច្បាប់ Big-O យើងលុបចោលមេគុណថេរ ទើបគេសន្និដ្ឋានថាវាជាល្បឿន O(n) ដដែល។

def find_pairs(arr):
    n = len(arr)

៣. ករណីសិក្សា៖ ពេលវេលាការ៉េ O(n²) Quadratic Time (ត)

មធ្យម ១ ក្បួនតម្រៀបទិន្នន័យ Bubble Sort

Copy & Run Code
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n - i - 1):
            # ប្រៀបធៀបអ្នកនៅក្បែរគ្នា បើអ្នកមុខធំជាង ដូរទីតាំង
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr

print("តម្រៀប:", bubble_sort([5, 1, 4, 2]))
ដោយសារវាត្រូវប្រៀបធៀបគូទិន្នន័យជាប់ៗគ្នាជាច្រើនជុំ (Loop ត្រួតគ្នា) ទើបវាជាក្បួនតម្រៀបដែលមានល្បឿនយឺតកម្រិត O(n²) មិនស័ក្តិសមជាមួយទិន្នន័យធំៗឡើយ។

ធ្ងន់ ១ រកមើលទិន្នន័យស្ទួនដោយប្រៀបធៀបគ្រប់គូ (Find Duplicates)

Copy & Run Code
def has_duplicates(arr):
    n = len(arr)
    for i in range(n):
        # ប្រៀបធៀប i ជាមួយធាតុបន្ទាប់ៗទាំងអស់
        for j in range(i + 1, n): 
            if arr[i] == arr[j]:
                return True
    return False

print("មានស្ទួនទេ?", has_duplicates([1, 2, 3, 1]))
វិធីនេះប្រៀបធៀបធាតុនីមួយៗជាមួយធាតុដទៃទៀតទាំងអស់។ ទោះបី Loop ក្នុងរត់មិនដល់ N ពេញ (រត់តែ N-i ដង) ក៏សរុបរួមតាមច្បាប់គណិតវិទ្យាវានៅតែស្ថិតក្នុងល្បឿន O(n²) ដដែល។

def print_halves(n):
    while n > 1:
        print("តម្លៃបច្ចុប្បន្ន:", n)

ស្រាល ២ ការស្វែងរកបែបពុះពាក់កណ្តាល (Binary Search)

def binary_search(arr, target):
    l, r = 0, len(arr) - 1
    while l <= r:
        mid = (l + r) // 2
        if arr[mid] == target: return mid
        elif arr[mid] < target: l = mid + 1
        else: r = mid - 1
    return -1

print("រកឃើញលេខ 4 នៅទីតាំង Index:", binary_search([1, 2, 3, 4, 5], 4))

def sqrt_bs(n):
    l, r = 0, n
    ans = 0
    while l <= r:
        mid = (l + r) // 2
        if mid * mid <= n:
            ans = mid  # រក្សាទុកតម្លៃដែលទំនងអាចជាចម្លើយ
            l = mid + 1
        else:
            r = mid - 1
    return ans

print("ឫសការ៉េនៃ 25 គឺ:", sqrt_bs(25))

def fast_power(base, exp):
    res = 1
    while exp > 0:
        if exp % 2 == 1: # បើស្វ័យគុណសេស
            res *= base
        base *= base       # ការ៉េខ្លួនឯង
        exp //= 2          # កាត់បន្ថយពាក់កណ្តាល
    return res

print("2 ស្វ័យគុណ 10 គឺ:", fast_power(2, 10))

def sum_O1_space(n):
    total = 0 # បង្កើតអថេរតែមួយគត់

def find_intersection(list1, list2):
    # បំប្លែងទៅជា Set ប្រើ Space O(n)
    set1 = set(list1) 
    
    # ស្វែងរក 'x in set1' ប្រើពេលតែ O(1) លឿនបំផុត
    res = [x for x in list2 if x in set1] 
    return res

print("តម្លៃដូចគ្នា:", find_intersection([1,2,3], [2,3,4]))
