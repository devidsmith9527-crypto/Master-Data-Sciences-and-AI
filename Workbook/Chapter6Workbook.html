class Graph:
    def __init__(self):
        self.adj_list = {}
    def add_edge(self, u, v):
        if u not in self.adj_list: self.adj_list[u] = []
        if v not in self.adj_list: self.adj_list[v] = []
        self.adj_list[u].append(v)
        self.adj_list[v].append(u) # មានទៅមានមក (Undirected)

g = Graph()
g.add_edge("A", "B")
print("Adjacency List:", g.adj_list)
យើងប្រើ Dictionary ដើម្បីផ្ទុក Graph។ Key ជាឈ្មោះចំណុច (Vertex) រីឯ Value ជាបញ្ជីឈ្មោះមិត្តភក្តិរបស់វា។ កូដនេះជា Undirected Graph ទើប A ស្គាល់ B ហើយ B ក៏ស្គាល់ A វិញ។

def build_graph(edges):
    graph = {}
    for u, v in edges:
        if u not in graph: graph[u] = []
        if v not in graph: graph[v] = []
        graph[u].append(v)
        graph[v].append(u)
    return graph

edges_data = [[0, 1], [0, 2], [1, 2]]
print("Graph ថ្មី:", build_graph(edges_data))

២. ករណីសិក្សា៖ ការរុករកក្នុង Graph (BFS)

ស្រាល ៣ ការរុករកបែបវាលរាប BFS (Breadth-First Search)

Copy & Run Code
from collections import deque

def bfs(graph, start):
    visited = set([start]) # កត់ចំណាំអ្នកដែលបានដើរដល់
    queue = deque([start]) # ប្រើ Queue ដើម្បីរៀបជួរអ្នកត្រូវដើរបន្ត
    
    while queue:
        node = queue.popleft()
        print(node, end=" ")
        
        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)

adj = {'A': ['B', 'C'], 'B': ['A', 'D'], 'C': ['A', 'D'], 'D': ['B', 'C']}
print("BFS លទ្ធផល:"); bfs(adj, 'A')
BFS ដើររុករកអ្នកជិតខាង (មិត្តភក្តិផ្ទាល់) ឱ្យអស់សិន ទើបទៅរុករកមិត្តរបស់មិត្តបន្តទៀត (រុករកជាជាន់ៗ) ដោយប្រើប្រាស់រចនាសម្ព័ន្ធ Queue។ ល្អសម្រាប់រកផ្លូវជិតបំផុត។

មធ្យម ២ រកផ្លូវខ្លីបំផុត (Shortest Path in Unweighted Graph)

Copy & Run Code
def shortest_path_length(graph, start, end):
    q = deque([(start, 0)]) # (Node បច្ចុប្បន្ន, ចម្ងាយរាប់ពីដើម)
    visited = set([start])
    
    while q:
        node, dist = q.popleft()
        if node == end: return dist
        
        for neighbor in graph.get(node, []):
            if neighbor not in visited:
                visited.add(neighbor)
                q.append((neighbor, dist + 1))
    return -1

g = {'A':['B','C'], 'B':['D'], 'C':['D','E'], 'D':['E']}
print("ចម្ងាយខ្លីបំផុតពី A ទៅ E គឺ:", shortest_path_length(g, 'A', 'E'))
សម្រាប់ Graph ដែលគ្មានគិតគីឡូម៉ែត្រ (Unweighted) ការប្រើ BFS (Level-order) គឺធានាថារកឃើញផ្លូវដែលឆ្លងកាត់ចំណុចតិចបំផុតជានិច្ច ព្រោះវាដើរពង្រីកជាវង់ៗ។

២. ករណីសិក្សា៖ ការរុករកក្នុង Graph (DFS)

ស្រាល ៤ ការរុករកបែបស៊ីជម្រៅ DFS (Depth-First Search)

Copy & Run Code
def dfs(graph, node, visited=None):
    if visited is None: visited = set()
    visited.add(node)
    print(node, end=" ")
    
    for neighbor in graph[node]:
        if neighbor not in visited:
            dfs(graph, neighbor, visited) # Recursive ចុះឱ្យជ្រៅ

adj = {'A': ['B', 'C'], 'B': ['A', 'D'], 'C': ['A', 'D'], 'D': ['B', 'C']}
print("DFS លទ្ធផល:"); dfs(adj, 'A')
DFS ដើររុករកតាមខ្សែមួយឱ្យដល់ផ្លូវទាល់សិន ទើបថយក្រោយ (Backtrack) មកបំបែកផ្លូវផ្សេង។ វាប្រើប្រាស់ Call Stack របស់កុំព្យូទ័រតាមរយៈ Recursive។ ល្អសម្រាប់ដោះស្រាយល្បែងវង្កត់ (Maze)។

មធ្យម ៣ រាវរកផ្លូវ (Path Exists) រវាង Node ពីរ

Copy & Run Code
def has_path(graph, src, dst, visited):
    if src == dst: return True
    visited.add(src)
    for neighbor in graph[src]:
        if neighbor not in visited:
            if has_path(graph, neighbor, dst, visited):
                return True
    return False

print("តើអាចដើរពី A ទៅ D បានទេ?", has_path(adj, 'A', 'D', set()))

def count_components(graph):
    visited = set()
    count = 0
    for node in graph:
        if node not in visited:
            dfs(graph, node, visited) # អូសយកអ្នកជាប់ទាក់ទងទាំងអស់
            count += 1 # រាប់ជាបណ្តាញ ១
    return count

# បណ្តាញ A,B ភ្ជាប់គ្នា និង C,D ភ្ជាប់គ្នា តែអត់ភ្ជាប់ A-C ទេ
isolated_graph = {'A':['B'], 'B':['A'], 'C':['D'], 'D':['C']}
print("\nចំនួនក្រុមសរុប:", count_components(isolated_graph))

def has_cycle(graph, node, visited, parent):
    visited.add(node)
    for neighbor in graph[node]:
        if neighbor not in visited:
            if has_cycle(graph, neighbor, visited, node): return True
        elif neighbor != parent: # បើធ្លាប់ដើរដល់ហើយ តែវាមិនមែនជាអ្នកហៅយើងមក
            return True # មានន័យថាវាជារង្វិលជុំ
    return False

g_cycle = {'A':['B','C'], 'B':['A','D'], 'C':['A','D'], 'D':['B','C']}
print("តើមាន Cycle ទេ?", has_cycle(g_cycle, 'A', set(), None))
យើងប្រើ DFS ដើរទៅមុខ។ បើយើងដើរទៅប៉ះចំចំណុចមួយដែលយើងធ្លាប់ដើរឆ្លងកាត់ម្តងរួចមកហើយ (ក្រៅពីចំណុចមេផ្ទាល់ដែលហៅយើងមក) នោះវាសន្និដ្ឋានបានថាបណ្តាញនោះមានរង្វិលជុំ (Cycle) ជាមិនខាន។

def two_sum(nums, target):
    seen = {} # ទុកលេខដែលបានឃើញ និងទីតាំងរបស់វា {តម្លៃ: Index}
    for i, num in enumerate(nums):
        needed = target - num
        if needed in seen:
            return [seen[needed], i]
        seen[num] = i
    return []

print("ទីតាំងលេខបូកស្មើ 9:", two_sum([2, 7, 11, 15], 9))

from collections import defaultdict

def group_anagrams(strs):
    groups = defaultdict(list) # Dictionary ដែលមាន Value ជា List ទទេស្វ័យប្រវត្តិ
    for word in strs:
        key = "".join(sorted(word)) # តម្រៀបអក្សរធ្វើជា Key (ឧ. "ate" -> "aet")
        groups[key].append(word)
    return list(groups.values())

print("ក្រុម:", group_anagrams(["eat", "tea", "tan", "ate", "nat", "bat"]))

def find_duplicate(nums):
    seen = set()
    for num in nums:
        if num in seen:
            return num
        seen.add(num)
    return None
