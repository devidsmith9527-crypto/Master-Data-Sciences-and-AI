ស្រាល ១ លេខ Fibonacci ដោយប្រើការចងចាំ (Memoization)

Copy & Run Code
# Memoization (Top-Down DP) ជួយសន្សំសំចៃពេលគណនាឡើងវិញ
def fib_memo(n, memo={}):
    if n in memo:
        return memo[n]  # ទាញយកចម្លើយដែលគណនារួចចេញមក
    if n <= 1:
        return n
    # បើមិនទាន់មាន គណនា រួចរក្សាទុកក្នុង memo
    memo[n] = fib_memo(n-1, memo) + fib_memo(n-2, memo)
    return memo[n]

print("Fibonacci ទី 50 គឺ:", fib_memo(50))
បើយើងប្រើ Recursion ធម្មតា ការរកលេខ F(50) ត្រូវចាយពេលរាប់ឆ្នាំ ព្រោះ Time Complexity O(2^n)។ តែពេលប្រើ Memoization (រក្សាទុកចម្លើយចាស់ក្នុង Dictionary) ល្បឿនធ្លាក់មកសល់ត្រឹម O(n) ចំណាយពេលតែមួយប៉ប្រិចភ្នែកប៉ុណ្ណោះ។

# Tabulation (Bottom-Up DP) - រកចំនួនកាក់តិចបំផុតសម្រាប់អាប់លុយ
def min_coins(coins, amount):
    # បង្កើត Array ផ្ទុកតម្លៃលុយនីមួយៗ ដំបូងដាក់តម្លៃអតិបរមា (Infinity)
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0  # លុយ 0៛ ត្រូវការកាក់ 0
    
    for a in range(1, amount + 1):
        for c in coins:
            if a - c >= 0:
                dp[a] = min(dp[a], 1 + dp[a - c])
                
    return dp[amount] if dp[amount] != float('inf') else -1

print("ត្រូវការកាក់តិចបំផុត:", min_coins([1, 5, 10, 25], 37))
នេះគឺជាបញ្ហា Optimization បុរាណ។ DP នឹងបំបែកបញ្ហាអាប់លុយ ៣៧ រៀល ទៅជាការអាប់លុយតូចៗចាប់ពី ១ រៀលឡើងទៅ (Bottom-up)។ ចម្លើយគឺត្រូវការកាក់ ៤ គ្រាប់ (25+10+1+1)។ ល្បឿនគឺ O(amount * len(coins))។

២. ការដោះស្រាយបញ្ហាធនធាន (The Knapsack Problem)

ធ្ងន់ ១ ជ្រើសរើសទំនិញដាក់កាបូបឱ្យចំណេញបំផុត (0/1 Knapsack)

Copy & Run Code
# វត្ថុនីមួយៗមានទម្ងន់ (wt) និងតម្លៃចំណេញ (val)។ កាបូបទ្រាំទ្របានទម្ងន់ W
def knapsack(W, wt, val, n):
    # បង្កើតតារាង DP ២ វិមាត្រ (2D Array) ផ្ទុកតម្លៃ 0
    K = [[0 for w in range(W + 1)] for i in range(n + 1)]

    # ដំណើរការបំពេញតារាង DP
    for i in range(n + 1):
        for w in range(W + 1):
            if i == 0 or w == 0:
                K[i][w] = 0
            elif wt[i-1] <= w: # យកតម្លៃធំបំផុតរវាង "ដាក់" និង "មិនដាក់"
                K[i][w] = max(val[i-1] + K[i-1][w-wt[i-1]],  K[i-1][w])
            else: # បើធ្ងន់ពេកដាក់មិនចូល យកតម្លៃចាស់
                K[i][w] = K[i-1][w]
    return K[n][W]

val, wt, W = [60, 100, 120], [10, 20, 30], 50
print("ប្រាក់ចំណេញអតិបរមាដែលអាចប្រមូលបានគឺ:", knapsack(W, wt, val, len(val)))
Knapsack Problem ជាគ្រឹះនៃ Resource Allocation។ យើងប្រើវាដើម្បីជ្រើសរើសវិនិយោគគម្រោងណាដែលចាយទុន (Weight) តិច តែបានប្រាក់ចំណេញ (Value) ខ្ពស់បំផុតក្នុងកញ្ចប់ថវិកាកំណត់។

import math

def euclidean_distance(p1, p2):
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def simple_knn(data_points, new_point, k):
    distances = []
    for point, label in data_points:
        distances.append((euclidean_distance(point, new_point), label))
    
    # តម្រៀបពីជិតជាងគេ ទៅឆ្ងាយជាងគេ O(n log n)
    distances.sort(key=lambda x: x[0])
    
    k_nearest = distances[:k]
    print(f"{k} នាក់ជិតបំផុតគឺ:", k_nearest)
    return "ចាត់ថ្នាក់រួចរាល់"

dataset = [((1, 2), 'A'), ((2, 3), 'A'), ((8, 8), 'B'), ((9, 10), 'B')]
simple_knn(dataset, (3, 4), k=2)
KNN គណនាចម្ងាយ (Distance) ទៅកាន់ទិន្នន័យចាស់ៗ។ បើចំណុចថ្មីនៅជិតក្រុម 'A' ច្រើនជាង វាសន្និដ្ឋានថាជារបស់ក្រុម 'A' ដែរ។ ប្រើញឹកញាប់ក្នុងការធ្វើ Recommendation Systems។

៤. ការស្វែងរកផ្លូវក្នុង AI (Pathfinding in AI)

ធ្ងន់ ២ ក្បួនស្វែងរកផ្លូវខ្លីបំផុត Dijkstra's Algorithm

Copy & Run Code
import heapq

def dijkstra(graph, start):
    # កំណត់ចម្ងាយទៅគ្រប់ Node ទាំងអស់ថាអនន្ត (Infinity) ជាមុនសិន
    distances = {node: float('infinity') for node in graph}
    distances[start] = 0 # ចម្ងាយពីដើមទីទៅខ្លួនឯងគឺ ០
    
    # ប្រើប្រាស់ Priority Queue (Min-Heap) ដើម្បីរើសផ្លូវដែលជិតជាងគេមុន O(E log V)
    priority_queue = [(0, start)]
    
    while priority_queue:
        # ទាញយក Node ដែលនៅជិតបំផុតមកត្រួតពិនិត្យ
        current_distance, current_node = heapq.heappop(priority_queue)
        
        # បើចម្ងាយក្នុងប្រព័ន្ធធំជាងចម្ងាយដែលរក្សាទុក យើងបោះបង់ចោលចម្ងាយចាស់
        if current_distance > distances[current_node]:
            continue
            
        # ពិនិត្យមើលអ្នកជិតខាងរបស់វាទាំងអស់ (Neighbors)
        for neighbor, weight in graph[current_node].items():
            distance = current_distance + weight
            
            # បើរកឃើញផ្លូវដែលជិតជាងផ្លូវចាស់ យើងធ្វើបច្ចុប្បន្នភាព
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                heapq.heappush(priority_queue, (distance, neighbor))
                
    return distances

# បង្កើតផែនទី Graph (ឧទាហរណ៍៖ ចម្ងាយគិតជាគីឡូម៉ែត្ររវាងខេត្ត)
khmer_map = {
    'SRP': {'BTB': 75, 'KPC': 200},
    'BTB': {'SRP': 75, 'PNH': 290},
    'KPC': {'SRP': 200, 'PNH': 120},
    'PNH': {'BTB': 290, 'KPC': 120}
}

# ស្វែងរកផ្លូវខ្លីបំផុតពីសៀមរាប ទៅគ្រប់ខេត្ត
shortest_paths = dijkstra(khmer_map, 'SRP')

print("ចម្ងាយខ្លីបំផុតពី សៀមរាប (SRP) ទៅ៖")
for destination, dist in shortest_paths.items():
    print(f" -> ខេត្ត {destination} គឺ {dist} គីឡូម៉ែត្រ")
Dijkstra ជាក្បួនដ៏ល្បីល្បាញ ដែលត្រូវប្រើប្រាស់ក្នុងប្រព័ន្ធ Google Maps ឬការ Routing Network។ វាប្រើប្រាស់ Min-Heap ដើម្បីធានាថាការស្វែងរកផ្លូវស្ថិតក្នុងជម្រើសដែលខាតបង់តិចបំផុតជានិច្ច ធ្វើឱ្យការងារចប់ក្នុងល្បឿន O(E log V)។
